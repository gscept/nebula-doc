368FixedArray<TYPE>::Resize(SizeT newSize)
369{
370 // allocate new array and copy over old elements
371 TYPE* newElements = 0;
372 if (newSize > 0)
373 {
374 newElements = new TYPE[newSize];
375 SizeT numCopy = this->count;
376 if (numCopy > 0)
377 {
378 if (numCopy > newSize)
379 numCopy = newSize;
380 if constexpr (!std::is_trivially_move_assignable<TYPE>::value && std::is_move_assignable<TYPE>::value)
381 {
382 IndexT i;
383 for (i = 0; i < numCopy; i++)
384 {
385 newElements[i] = std::move(this->elements[i]);
386 }
387 }
388 else
389 {
390 Memory::MoveElements(this->elements, newElements, numCopy);
391 }
392 }
393 }

395 // delete old elements
396 this->Delete();
397
398 // set content to new elements
399 this->elements = newElements;

534template<class TYPE> IndexT
535FixedArray<TYPE>::BinarySearchIndex(const TYPE& elm) const
536{
537 SizeT num = this->Size();
538 if (num > 0)
539 {
540 IndexT half;
541 IndexT lo = 0;
542 IndexT hi = num - 1;
543 IndexT mid;
544 while (lo <= hi)
545 {
546 if (0 != (half = num/2))
547 {
548 mid = lo + ((num & 1) ? half : (half - 1));
549 if (elm < this->elements[mid])
550 {
551 hi = mid - 1;
552 num = num & 1 ? half : half - 1;
553 }
554 else if (elm > this->elements[mid])
555 {
556 lo = mid + 1;
557 num = half;
558 }
559 else
560 {
561 return mid;
562 }
563 }
564 else if (num)
565 {
566 if (elm != this->elements[lo])
567 {
568 return InvalidIndex;
569 }
570 else
571 {
572 return lo;
573 }
574 }
575 else
576 {
577 break;
578 }
579 }
580 }
581 return InvalidIndex;
