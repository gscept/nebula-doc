226template <int ChunkSize>
227template<typename T>
228inline T*
229ArenaAllocator<ChunkSize>::Alloc()
230{
231 T* ret = nullptr;
232 if constexpr (sizeof(T) > ChunkSize)
233 {
234 // allocate block and retire it directly
235 byte* block = (byte*)Memory::Alloc(ObjectArrayHeap, sizeof(T));
236 this->retiredChunks.Append(block);
237 ret = new (block) T;
238 }
239 else
240 {
241 // pad up to next multiple of 16 to avoid alignment issues
242 SizeT alignedSize = Math::align(sizeof(T), 16);
243 if (this->iterator == nullptr)
244 {
245 this->NewChunk();
246 }
247 else
248 {
249 // we cast the pointer diff but it should be safe since it should never be above ChunkSize
250 SizeT remainder = ChunkSize - SizeT(this->iterator - this->currentChunk);
251 if (remainder < alignedSize)
252 this->NewChunk();
253 }
254
255 ret = new (this->iterator) T;
256 this->iterator += alignedSize;
257 }
258 return ret;
259}

264template <int ChunkSize>
265inline void*
266ArenaAllocator<ChunkSize>::Alloc(SizeT size)
267{
268 n_assert(size != 0);
269 void* ret = nullptr;
270 if (size > ChunkSize)
271 {
272 // allocate block and retire it directly
273 byte* block = (byte*)Memory::Alloc(ObjectArrayHeap, size);
274 this->retiredChunks.Append(block);
275 ret = block;
276 }
277 else
278 {
279 // pad to next alignment.
280 size = Math::align(size, 16);
281 if (this->iterator == nullptr)

284 }
285 else
286 {
287 PtrDiff remainder = this->currentChunk + ChunkSize - this->iterator;
288 if (remainder < size)
289 this->NewChunk();
290 }
291 ret = this->iterator;
