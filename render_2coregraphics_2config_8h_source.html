135//------------------------------------------------------------------------------
138inline ShaderVisibility
139ShaderVisibilityFromString(const Util::String& str)
140{
141 Util::Array<Util::String> components = str.Tokenize("|");
142 CoreGraphics::ShaderVisibility ret = InvalidVisibility;
143 IndexT i;
144 for (i = 0; i < components.Size(); i++)
145 {
146 const Util::String& component = components[i];
147 if (component == "VS") ret |= VertexShaderVisibility;
148 else if (component == "HS") ret |= HullShaderVisibility;
149 else if (component == "DS") ret |= DomainShaderVisibility;
150 else if (component == "GS") ret |= GeometryShaderVisibility;
151 else if (component == "PS") ret |= PixelShaderVisibility;
152 else if (component == "CS") ret |= ComputeShaderVisibility;
153 }
154
155 return ret;
156}
