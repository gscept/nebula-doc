122inline Resources::ResourceId
123Resources::ResourceServer::CreateResource(const ResourceName& res, const Util::StringAtom& tag, std::function<void(const Resources::ResourceId)> success, std::function<void(const Resources::ResourceId)> failed, bool immediate)
124{
125 // get resource loader by extension
126 Util::String ext = res.AsString().GetFileExtension();
127 IndexT i = this->extensionMap.FindIndex(ext);
128 n_assert_fmt(i != InvalidIndex, "No resource loader is associated with file extension '%s'", ext.AsCharPtr());
129 const Ptr<ResourceStreamCache>& loader = this->pools[this->extensionMap.ValueAtIndex(i)].downcast<ResourceStreamCache>();
130
131 // create container and cast to actual resource type
132 Resources::ResourceId id = loader->CreateResource(res, nullptr, 0, tag, success, failed, immediate);
133 return id;
134}
