399template<class TYPE, int STACK_SIZE> void
400ArrayStack<TYPE, STACK_SIZE>::operator=(const ArrayStack<TYPE, STACK_SIZE>& rhs)
401{
402 if (this != &rhs)
403 {
404 if ((this->capacity > 0) && (rhs.count <= this->capacity))
405 {
406 // source array fits into our capacity, copy in place
407 n_assert(0 != this->elements);
408 IndexT i;
409 for (i = 0; i < rhs.count; i++)
410 {
411 this->elements[i] = rhs.elements[i];
412 }
413
414 // properly destroy remaining original elements
415 for (; i < this->count; i++)
416 {
417 this->Destroy(&(this->elements[i]));
418 }
419 this->grow = rhs.grow;
420 this->count = rhs.count;
421 }
422 else
423 {
424 // source array doesn't fit into our capacity, need to reallocate
425 this->Delete();
426 this->Copy(rhs);
427 }
428 }

489//------------------------------------------------------------------------------
492template<class TYPE, int STACK_SIZE> void
493ArrayStack<TYPE, STACK_SIZE>::Grow()
494{
495 #if NEBULA_BOUNDSCHECKS
496 n_assert(this->grow > 0);
497 #endif
498
499 SizeT growToSize;
500 if (0 == this->capacity)
501 {
502 growToSize = this->grow;
503 }
504 else
505 {
506 // grow by half of the current capacity, but never more then MaxGrowSize
507 SizeT growBy = this->capacity >> 1;
508 if (growBy == 0)
509 {
510 growBy = MinGrowSize;
511 }
512 else if (growBy > MaxGrowSize)
513 {
514 growBy = MaxGrowSize;
515 }
516 growToSize = this->capacity + growBy;
517 }
518 this->GrowTo(growToSize);
519}

521//------------------------------------------------------------------------------
526template<class TYPE, int STACK_SIZE> void
527ArrayStack<TYPE, STACK_SIZE>::Move(IndexT fromIndex, IndexT toIndex)
528{
529 #if NEBULA_BOUNDSCHECKS
530 n_assert(this->elements);
531 n_assert(fromIndex < this->count);
532 #endif
533
534 // nothing to move?
535 if (fromIndex == toIndex)
536 {
537 return;
538 }
539
540 // compute number of elements to move
541 SizeT num = this->count - fromIndex;
542
543 // check if array needs to grow
544 SizeT neededSize = toIndex + num;
545 while (neededSize > this->capacity)
546 {
547 this->Grow();
548 }
549
550 if (fromIndex > toIndex)
551 {
552 // this is a backward move
553 IndexT i;
554 for (i = 0; i < num; i++)
555 {
556 this->elements[toIndex + i] = this->elements[fromIndex + i];
557 }
558
559 // destroy remaining elements
560 for (i = (fromIndex + i) - 1; i < this->count; i++)
561 {
562 this->Destroy(&(this->elements[i]));
563 }
564 }
565 else
566 {
567 // this is a forward move
568 IndexT i; // NOTE: this must remain signed for the following loop to work!!!
569 for (i = num - 1; i >= 0; --i)
570 {
571 this->elements[toIndex + i] = this->elements[fromIndex + i];
572 }
573
574 // destroy freed elements
575 for (i = int(fromIndex); i < int(toIndex); i++)
576 {
577 this->Destroy(&(this->elements[i]));
578 }
579 }
580
581 // adjust array size
582 this->count = toIndex + num;
583}

1297template<class TYPE, int STACK_SIZE> IndexT
1298ArrayStack<TYPE, STACK_SIZE>::InsertSorted(const TYPE& elm)
1299{
1300 SizeT num = this->Size();
1301 if (num == 0)
1302 {
1303 // array is currently empty
1304 this->Append(elm);
1305 return this->Size() - 1;
1306 }
1307 else
1308 {
1309 IndexT half;
1310 IndexT lo = 0;
1311 IndexT hi = num - 1;
1312 IndexT mid;
1313 while (lo <= hi)
1314 {
1315 if (0 != (half = num/2))
1316 {
1317 mid = lo + ((num & 1) ? half : (half - 1));
1318 if (elm < this->elements[mid])
1319 {
1320 hi = mid - 1;
1321 num = num & 1 ? half : half - 1;
1322 }
1323 else if (elm > this->elements[mid])
1324 {
1325 lo = mid + 1;
1326 num = half;
1327 }
1328 else
1329 {
1330 // element already exists at [mid], append the
1331 // new element to the end of the range
1332 return this->InsertAtEndOfIdenticalRange(mid, elm);
1333 }
1334 }
1335 else if (0 != num)
1336 {
1337 if (elm < this->elements[lo])
1338 {
1339 this->Insert(lo, elm);
1340 return lo;
1341 }
1342 else if (elm > this->elements[lo])
1343 {
1344 this->Insert(lo + 1, elm);
1345 return lo + 1;
1346 }
1347 else
1348 {
1349 // element already exists at [low], append
1350 // the new element to the end of the range
1351 return this->InsertAtEndOfIdenticalRange(lo, elm);
1352 }
1353 }
1354 else
1355 {
1356 #if NEBULA_BOUNDSCHECKS
1357 n_assert(0 == lo);
1358 #endif
1359 this->Insert(lo, elm);
1360 return lo;
1361 }
1362 }
1363 if (elm < this->elements[lo])
1364 {
1365 this->Insert(lo, elm);
1366 return lo;
1367 }
1368 else if (elm > this->elements[lo])
1369 {
1370 this->Insert(lo + 1, elm);
1371 return lo + 1;
1372 }
1373 else
1374 {
1375 // can't happen(?)
1376 }
1377 }
1378 // can't happen
1379 n_error("Array::InsertSorted: Can't happen!");
1380 return InvalidIndex;
