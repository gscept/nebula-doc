483inline uint32_t
484Blob::HashCode() const
485{
486 IndexT hash = 0;
487 const char* charPtr = (const char*) this->ptr;
488 IndexT i;
489 for (i = 0; i < this->size; i++)
490 {
491 hash += charPtr[i];
492 hash += hash << 10;
493 hash ^= hash >> 6;
494 }
495 hash += hash << 3;
496 hash ^= hash >> 11;
497 hash += hash << 15;
498 hash &= ~(1<<31); // don't return a negative number (in case IndexT is defined signed)
499 return hash;
500}
