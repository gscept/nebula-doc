96inline
97void
98TableSignature::Setup(AttributeId const* componentBuffer, SizeT num)
99{
100 if (num > 0)
101 {
102 Util::Array<AttributeId> descriptors(componentBuffer, num);
103 descriptors.Sort();
104 AttributeId const last = *(descriptors.End() - 1);
105 n_assert(last != AttributeId::Invalid());
106 uint const largestBit = last.id;
107 this->size = (largestBit / 128) + 1;
108 this->mask = (__m128i*)Memory::Alloc(Memory::HeapType::ObjectHeap, this->size * 16);
109
110 for (int i = 0; i < this->size; i++)
111 this->mask[i] = _mm_setzero_si128();
112
113 alignas(16) uint64_t partialMask[2] = { 0, 0 };
114 // offsets start at first values offset
115 uint64_t offset = descriptors[0].id / 128;
116 uint64_t prevOffset = offset;
117 for (AttributeId i : descriptors)
118 {
119 n_assert(i != AttributeId::Invalid());
120 offset = i.id / 128;
121 if (offset != prevOffset)
122 {
123 // Write value and reset partial mask
124 *(this->mask + prevOffset) = _mm_set_epi64x(partialMask[0], partialMask[1]);
125 prevOffset = offset;
126 memset(&partialMask, 0, sizeof(partialMask));
127 }
128 uint32_t mod = (i.id % 128);
129 partialMask[mod / 64] |= 1ull << mod;
130 }
131
132 *(this->mask + offset) = _mm_set_epi64x(partialMask[0], partialMask[1]);
133 }
134 else
135 {
136 this->mask = nullptr;
137 this->size = 0;
138 }
139}

196TableSignature::IsSet(AttributeId component) const
197{
198 int offset = component.id / 128;
199 if (offset < this->size)
200 {
201 alignas(16) uint64_t partialMask[2] = { 0, 0 };
202 uint64_t const bit = component.id % 128;
203 partialMask[bit / 64] |= 1ull << bit;
204 __m128i temp = _mm_set_epi64x(partialMask[0], partialMask[1]);
205 int const isSet = !_mm_testz_si128(temp, this->mask[offset]);

253TableSignature::CheckBits(TableSignature const& src, TableSignature const& mask)
254{
255 if (src.size == 0 || mask.size == 0)
256 return false;
257
258#ifdef NEBULA_DEBUG
