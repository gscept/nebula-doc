174inline CoreGraphics::ImageAspect
175ImageAspectFromString(const Util::String& str)
176{
177 Util::Array<Util::String> comps = str.Tokenize("|");
178 CoreGraphics::ImageAspect aspect = CoreGraphics::ImageAspect(0x0);
179 for (IndexT i = 0; i < comps.Size(); i++)
180 {
181 if (comps[i] == "Color") aspect |= CoreGraphics::ImageAspect::ColorBits;
182 else if (comps[i] == "Depth") aspect |= CoreGraphics::ImageAspect::DepthBits;
183 else if (comps[i] == "Stencil") aspect |= CoreGraphics::ImageAspect::StencilBits;
184 else if (comps[i] == "Metadata") aspect |= CoreGraphics::ImageAspect::MetaBits;
185 else if (comps[i] == "Plane0") aspect |= CoreGraphics::ImageAspect::Plane0Bits;
186 else if (comps[i] == "Plane1") aspect |= CoreGraphics::ImageAspect::Plane1Bits;
187 else if (comps[i] == "Plane2") aspect |= CoreGraphics::ImageAspect::Plane2Bits;
188 else
189 {
190 n_error("Invalid access string '%s'\n", comps[i].AsCharPtr());
191 return CoreGraphics::ImageAspect::ColorBits;
192 }
193 }
194 return aspect;
195}
