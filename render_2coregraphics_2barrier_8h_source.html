195inline CoreGraphics::ImageBits
196ImageBitsFromString(const Util::String& str)
197{
198 Util::Array<Util::String> comps = str.Tokenize("|");
199 CoreGraphics::ImageBits bits = CoreGraphics::ImageBits(0x0);
200 for (IndexT i = 0; i < comps.Size(); i++)
201 {
202 if (comps[i] == "Auto") { bits = CoreGraphics::ImageBits::Auto; break; }
203 else if (comps[i] == "Color") bits |= CoreGraphics::ImageBits::ColorBits;
204 else if (comps[i] == "Depth") bits |= CoreGraphics::ImageBits::DepthBits;
205 else if (comps[i] == "Stencil") bits |= CoreGraphics::ImageBits::StencilBits;
206 else if (comps[i] == "Metadata") bits |= CoreGraphics::ImageBits::MetaBits;
207 else if (comps[i] == "Plane0") bits |= CoreGraphics::ImageBits::Plane0Bits;
208 else if (comps[i] == "Plane1") bits |= CoreGraphics::ImageBits::Plane1Bits;
209 else if (comps[i] == "Plane2") bits |= CoreGraphics::ImageBits::Plane2Bits;
210
211 else
212 {
213 n_error("Invalid access string '%s'\n", comps[i].AsCharPtr());
214 return CoreGraphics::ImageBits::ColorBits;
215 }
216 }
217 return bits;
218}

223inline CoreGraphics::ImageLayout
224ImageLayoutFromString(const Util::String& str)
225{
226 if (str == "Undefined") return CoreGraphics::ImageLayout::Undefined;
227 else if (str == "General") return CoreGraphics::ImageLayout::General;
228 else if (str == "ColorRenderTexture") return CoreGraphics::ImageLayout::ColorRenderTexture;
229 else if (str == "DepthRenderTexture") return CoreGraphics::ImageLayout::DepthStencilRenderTexture;
230 else if (str == "DepthStencilRead") return CoreGraphics::ImageLayout::DepthStencilRead;
231 else if (str == "ShaderRead") return CoreGraphics::ImageLayout::ShaderRead;
232 else if (str == "TransferSource") return CoreGraphics::ImageLayout::TransferSource;
233 else if (str == "TransferDestination") return CoreGraphics::ImageLayout::TransferDestination;
234 else if (str == "Preinitialized") return CoreGraphics::ImageLayout::Preinitialized;
235 else if (str == "Present") return CoreGraphics::ImageLayout::Present;
236 return CoreGraphics::ImageLayout::Undefined;
