179template<typename T>
180inline bool
181JsonReader::GetOpt(T& target, const char* attr)
182{
183 if (this->HasAttr(attr))
184 {
185 this->Get(target, attr);
186 return true;
187 }
188 return false;
189}

191//------------------------------------------------------------------------------
194template<typename T>
195inline bool
196JsonReader::GetOpt(T& target, const char* attr, const T& defaultVal)
197{
198 if (!this->GetOpt<T>(target, attr))
199 {
200 target = defaultVal;
201 return false;
202 }
203 return true;
204}

209template<unsigned int N>
210inline void
211JsonReader::Get(Util::BitField<N>& ret, const char* attr)
212{
213 Util::Array<int> arr;
214 this->Get<Util::Array<int>>(arr);
215
216 unsigned int count = arr.Size();
217 n_assert(count <= N);
218 for (unsigned int i = 0; i < count; i++)
219 {
220 ret.SetBit(arr[i]);
221 }
222}
