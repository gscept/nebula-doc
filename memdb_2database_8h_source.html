140inline void*
141Database::GetValuePointer(TableId table, ColumnIndex cid, IndexT row)
142{
143 n_assert(cid != ColumnIndex::Invalid());
144 n_assert(this->IsValid(table));
145 Table& tbl = this->tables[Ids::Index(table.id)];
146 ComponentId descriptor = tbl.columns.Get<0>(cid.id);
147 ComponentDescription* desc = MemDb::TypeRegistry::GetDescription(descriptor);
148 if (desc->typeSize > 0)
149 return ((byte*)tbl.columns.Get<1>(cid.id)) + (desc->typeSize * row);
150
151 return nullptr;
152}
