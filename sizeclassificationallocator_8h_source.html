151inline void
152SizeClassificationAllocator::Clear()
153{
154 // Reset bit masks
155 this->bucketUsageMask = 0x0;
156 memset(this->binMasks, 0x0, sizeof(this->binMasks));
157
158 // Reset bin heads
159 for (uint i = 0; i < NUM_BUCKETS * NUM_BINS_PER_BUCKET; i++)
160 {
161 this->binHeads[i] = SCNode::END;
162 }
163
164 // Clear nodes
165 for (uint i = 0; i < this->nodes.Size(); i++)
166 {
167 this->freeNodes[i] = this->nodes.Size() - i - 1;
168 }
169 this->freeNodeIterator = this->nodes.Size() - 1;
170 this->freeStorage = 0;
171
172 // Insert node covering the whole range
173 if (this->size > 0)
174 this->InsertNode(this->size, 0);
175}

189inline SizeClassificationAllocation
190SizeClassificationAllocator::Alloc(uint size, uint alignment)
191{
192 // We are not allowed any more allocations
193 if (this->freeStorage < size)
194 {
195 return { SCAlloc::OOM, SCAlloc::OOM };
196 }
197
198 BinIndex minIndex = IndexFromSize(size);
199
200 uint bin = 0xFF;
201 uint bucket = minIndex.bucket;
202
203 if (this->bucketUsageMask & (1 << bucket))
204 {
205 bin = Lsb(this->binMasks[bucket], minIndex.bin);
206 }
207
208 if (bin > NUM_BINS_PER_BUCKET)
209 {
210 // Find next bit set after bucket in the usage mask
211 bucket = Lsb(this->bucketUsageMask, bucket + 1);
212
213 // If this means we get 32, it means we're out of buckets that fit
214 if (bucket > NUM_BUCKETS)
215 {
216 return { SCAlloc::OOM, SCAlloc::OOM };
217 }
218
219 // Find any bit, since this bucket has to fit
220 bin = FirstOne(this->binMasks[bucket]);
221 }
222
223 BinIndex binIndex;
224 binIndex.bucket = bucket;
225 binIndex.bin = bin;
226
227 // Get linked list head
228 uint nodeIndex = this->binHeads[binIndex.index];
229 SCNode& node = this->nodes[nodeIndex];
230
231 // Calculate padding required by alignment
232 uint padding = Math::align(node.offset, alignment) - node.offset;
233
234 // Since we get a new size, need to check there is space with padding
235 if (this->freeStorage < (size + padding))
236 {
237 return { SCAlloc::OOM, SCAlloc::OOM };
238 }
239
240 // Save total size of node
241 uint totalSize = node.size;
242 uint baseOffset = node.offset;
243 node.size = size;
244 node.resident = true;
245
246 // Offset by padding
247 node.offset += padding;
248
249 // Bump head of bin to next node
250 this->binHeads[binIndex.index] = node.binNext;
251 if (node.binNext != SCNode::END)
252 this->nodes[node.binNext].binPrev = SCNode::END;
253
254 // If bin head is empty after we grab the node, unmark the bin
255 if (this->binHeads[binIndex.index] == SCNode::END)
256 {
257 this->binMasks[binIndex.bucket] &= ~(1 << binIndex.bin);
258
259 // If all bins are busy, unmark the bucket bits
260 if (this->binMasks[binIndex.bucket] == 0)
261 {
262 this->bucketUsageMask &= ~(1 << binIndex.bucket);
263 }
264 }
265
266 // We subtract the entire size of the node
267 this->freeStorage -= totalSize;
268
269 // The remainder in the front is then added back
270 uint remainder = totalSize - padding - node.size;
271 if (remainder > 0)
272 {
273 uint newNodeIndex = this->InsertNode(remainder, node.offset + size);
274 SCNode& newNode = this->nodes[newNodeIndex];
275
276 // If the current node has a next, repoint it to the new block
277 if (node.blockNext != SCNode::END)
278 {
279 this->nodes[node.blockNext].blockPrev = newNodeIndex;
280 }
281 newNode.blockPrev = nodeIndex;
282 newNode.blockNext = node.blockNext;
283 node.blockNext = newNodeIndex;
284 }
285
286 // If there was any padding offset, that memory is also added back
287 if (padding > 0)
288 {
289 uint newNodeIndex = this->InsertNode(padding, baseOffset);
290 SCNode& newNode = this->nodes[newNodeIndex];
291
292 if (node.blockPrev != SCNode::END)
293 {
294 this->nodes[node.blockPrev].blockNext = newNodeIndex;
295 }
296 if (node.blockNext != SCNode::END)
297 {
298 this->nodes[node.blockNext].blockPrev = newNodeIndex;
299 }
300 newNode.blockPrev = node.blockPrev;
301 newNode.blockNext = nodeIndex;
302 node.blockPrev = newNodeIndex;
303 }
304
305 return { node.offset, nodeIndex };
306}

311inline void
312SizeClassificationAllocator::Dealloc(const SizeClassificationAllocation& allocation)
313{
314 SCNode& node = this->nodes[allocation.node];
315 n_assert(node.resident);
316 node.resident = false;
317
318 uint size = node.size;
319 uint offset = node.offset;
320
321 // Try to merge with left
322 if (node.blockPrev != SCNode::END)
323 {
324 SCNode& prev = this->nodes[node.blockPrev];
325 if (!prev.resident)
326 {
327 offset = prev.offset;
328 size += prev.size;
329
330 this->RemoveNode(node.blockPrev);
331 node.blockPrev = prev.blockPrev;
332 }
333 }
334
335 if (node.blockNext != SCNode::END)
336 {
337 SCNode& next = this->nodes[node.blockNext];
338 if (!next.resident)
339 {
340 size += next.size;
341 this->RemoveNode(node.blockNext);
342 node.blockNext = next.blockNext;
343 }
344 }
345
346 uint leftNode = node.blockPrev;
347 uint rightNode = node.blockNext;
348 this->freeNodes[++this->freeNodeIterator] = allocation.node;
349
350 // Add merged node
351 uint mergedNode = this->InsertNode(size, offset);
352
353 // Connect adjacent nodes with newly inserted node
354 if (leftNode != SCNode::END)
355 {
356 this->nodes[leftNode].blockNext = mergedNode;
357 this->nodes[mergedNode].blockPrev = leftNode;
358 }
359
360 if (rightNode != SCNode::END)
361 {
362 this->nodes[rightNode].blockPrev = mergedNode;
363 this->nodes[mergedNode].blockNext = rightNode;
364 }
365}

370inline uint
371SizeClassificationAllocator::InsertNode(uint size, uint offset)
372{
373 BinIndex index = IndexFromSize(size);
374
375 if (this->binHeads[index.index] == SCNode::END)
376 {
377 // Bin isn't used, flip bit for bucket and bin
378 this->binMasks[index.bucket] |= (1 << index.bin);
379 this->bucketUsageMask |= (1 << index.bucket);
380 }
381
382 // Get new node
383 uint headNodeIndex = this->binHeads[index.index];
384 uint newNodeIndex = this->freeNodes[this->freeNodeIterator--];
385 SCNode& newNode = this->nodes[newNodeIndex];
386
387 // Set the size and offset of the node memory
388 newNode.size = size;
389 newNode.offset = offset;
390 newNode.binNext = headNodeIndex;
391 newNode.binPrev = SCNode::END;
392
393 this->freeStorage += size;
394
395 // Get node at head of bin
396 if (headNodeIndex != SCNode::END)
397 {
398 // If valid, repoint head node to point to the new node in the bin
399 SCNode& headNode = this->nodes[headNodeIndex];
400 headNode.binPrev = newNodeIndex;
401 }
402 this->binHeads[index.index] = newNodeIndex;
403
404 return newNodeIndex;
405}

407//------------------------------------------------------------------------------
410inline void
411SizeClassificationAllocator::RemoveNode(uint nodeIndex)
412{
413 // Get node to free
414 SCNode& node = this->nodes[nodeIndex];
415 node.resident = false;
416
417 // Add back the storage to the allocator
418 this->freeStorage -= node.size;
419
420 if (node.binPrev == SCNode::END)
421 {
422 BinIndex index = IndexFromSize(node.size);
423
424 // Point head of
425 this->binHeads[index.index] = node.binNext;
426 if (node.binNext != SCNode::END)
427 {
428 this->nodes[node.binNext].binPrev = SCNode::END;
429 }
430
431 if (node.binNext == SCNode::END)
432 {
433 // If we are deleting the last node, make sure to unmark the bin
434 this->binMasks[index.bucket] &= ~(1 << index.bin);
435
436 if (this->binMasks[index.bucket] == 0x0)
437 {
438 // If all bins have been cleared, clear the entire bucket
439 this->bucketUsageMask &= ~(1 << index.bucket);
440 }
441 }
442 }
443 else
444 {
445 this->nodes[node.binPrev].binNext = node.binNext;
446 if (node.binNext != SCNode::END)
447 {
448 this->nodes[node.binNext].binPrev = node.binPrev;
449 }
450 }
451
452 this->freeNodes[++this->freeNodeIterator] = nodeIndex;
453}
