510//------------------------------------------------------------------------------
513template<class TYPE> void
514Array<TYPE>::Grow()
515{
516 #if NEBULA_BOUNDSCHECKS
517 n_assert(this->grow > 0);
518 #endif
519
520 SizeT growToSize;
521 if (0 == this->capacity)
522 {
523 growToSize = this->grow;
524 }
525 else
526 {
527 // grow by half of the current capacity, but never more then MaxGrowSize
528 SizeT growBy = this->capacity >> 1;
529 if (growBy == 0)
530 {
531 growBy = MinGrowSize;
532 }
533 else if (growBy > MaxGrowSize)
534 {
535 growBy = MaxGrowSize;
536 }
537 growToSize = this->capacity + growBy;
538 }
539 this->GrowTo(growToSize);
540}

542//------------------------------------------------------------------------------
547template<class TYPE> void
548Array<TYPE>::Move(IndexT fromIndex, IndexT toIndex)
549{
550 #if NEBULA_BOUNDSCHECKS
551 n_assert(this->elements);
552 n_assert(fromIndex < this->count);
553 #endif
554
555 // nothing to move?
556 if (fromIndex == toIndex)
557 {
558 return;
559 }
560
561 // compute number of elements to move
562 SizeT num = this->count - fromIndex;
563
564 // check if array needs to grow
565 SizeT neededSize = toIndex + num;
566 while (neededSize > this->capacity)
567 {
568 this->Grow();
569 }
570
571 if (fromIndex > toIndex)
572 {
573 // this is a backward move
574 this->MoveRange(&this->elements[toIndex], &this->elements[fromIndex], num);
575 this->DestroyRange(fromIndex + num - 1, this->count);
576 }
577 else
578 {
579 // this is a forward move
580 int i; // NOTE: this must remain signed for the following loop to work!!!
581 for (i = num - 1; i >= 0; --i)
582 {
583 this->elements[toIndex + i] = this->elements[fromIndex + i];
584 }
585
586 // destroy freed elements
587 this->DestroyRange(fromIndex, toIndex);
588 }
589
590 // adjust array size
591 this->count = toIndex + num;
592}
