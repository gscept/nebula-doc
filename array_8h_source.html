509template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED> void
510Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::operator=(const Array<TYPE, SMALL_VECTOR_SIZE, PINNED>& rhs)
511{
512 if (this != &rhs)
513 {
514 if ((this->capacity > 0) && (rhs.count <= this->capacity))
515 {
516 // source array fits into our capacity, copy in place
517 n_assert(0 != this->elements);
518
519 this->CopyRange(this->elements, rhs.elements, rhs.count);
520 if (rhs.count < this->count)
521 {
522 this->DestroyRange(rhs.count, this->count);
523 }
524 this->grow = rhs.grow;
525 this->count = rhs.count;
526 }
527 else
528 {
529 // source array doesn't fit into our capacity, need to reallocate
530 this->Delete();
531 this->Copy(rhs);
532 }

568//------------------------------------------------------------------------------
571template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED> void
572Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::GrowTo(SizeT newCapacity)
573{
574 if (newCapacity > SMALL_VECTOR_SIZE)
575 {
576 if constexpr (PINNED)
577 {
578 if (this->elements == nullptr)
579 {
580 SizeT pageSize = System::PageSize;
581 SizeT reservationSize = Math::align(this->maxCommitSize * sizeof(TYPE), pageSize);
582 this->elements = (TYPE*)Memory::AllocVirtual(reservationSize);
583 }
584
585 SizeT pageSize = System::PageSize;
586
587 // Total amount of bytes needed to fill new capacity
588 SizeT totalByteSize = newCapacity * sizeof(TYPE);
589
590 // Rounded up to the page size so we don't waste memory we allocate anyways
591 SizeT totalBytesNeeded = Math::align(totalByteSize, pageSize);
592 n_assert(totalBytesNeeded <= this->maxCommitSize);
593 SizeT roundedUpNewCapacity = totalBytesNeeded / sizeof(TYPE);
594 SizeT offset = this->capacity * sizeof(TYPE);
595 if (totalBytesNeeded > offset)
596 {
597 // The amount of bytes we need to commit is the difference between the new and the old capacity
598 SizeT commitSize = (roundedUpNewCapacity - this->capacity) * sizeof(TYPE);
599 this->capacity = roundedUpNewCapacity;
600 Memory::CommitVirtual(((byte*)this->elements) + offset, commitSize);
601 }
602 }
603 else
604 {
605 TYPE* newArray = ArrayAlloc<TYPE>(newCapacity);
606 if (this->elements)
607 {
608 this->MoveRange(newArray, this->elements, this->count);
609
610 // discard old array if not the stack array
611 if (this->elements != this->stackElements.data())
612 ArrayFree(this->capacity, this->elements);
613 }
614 this->elements = newArray;
615 this->capacity = newCapacity;
616 }
617 }
618}

620//------------------------------------------------------------------------------
623template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED>
624void
625Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::Grow()
626{
627 #if NEBULA_BOUNDSCHECKS
628 n_assert(this->grow > 0);
629 #endif
630
631 SizeT growToSize;
632 if (0 == this->capacity)
633 {
634 growToSize = this->grow;
635 }
636 else
637 {
638 // grow by half of the current capacity, but never more then MaxGrowSize
639 SizeT growBy = this->capacity >> 1;
640 if (growBy == 0)
641 {
642 growBy = MinGrowSize;
643 }
644 else if (growBy > MaxGrowSize)
645 {
646 growBy = MaxGrowSize;
647 }
648 growToSize = this->capacity + growBy;
649 }
650 this->GrowTo(growToSize);
651}

653//------------------------------------------------------------------------------
658template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED>
659void
660Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::Move(IndexT fromIndex, IndexT toIndex)
661{
662 #if NEBULA_BOUNDSCHECKS
663 n_assert(this->elements);
664 n_assert(fromIndex < this->count);
665 #endif
666
667 // nothing to move?
668 if (fromIndex == toIndex)
669 {
670 return;
671 }
672
673 // compute number of elements to move
674 SizeT num = this->count - fromIndex;
675
676 // check if array needs to grow
677 SizeT neededSize = toIndex + num;
678 while (neededSize > this->capacity)
679 {
680 this->Grow();
681 }
682
683 if (fromIndex > toIndex)
684 {
685 // this is a backward move
686 this->MoveRange(&this->elements[toIndex], &this->elements[fromIndex], num);
687 this->DestroyRange(fromIndex + num - 1, this->count);
688 }
689 else
690 {
691 // this is a forward move
692 int i; // NOTE: this must remain signed for the following loop to work!!!
693 for (i = num - 1; i >= 0; --i)
694 {
695 this->elements[toIndex + i] = this->elements[fromIndex + i];
696 }
697
698 // destroy freed elements
699 this->DestroyRange(fromIndex, toIndex);
700 }
701
702 // adjust array size
703 this->count = toIndex + num;
704}
