511template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED> void
512Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::operator=(const Array<TYPE, SMALL_VECTOR_SIZE, PINNED>& rhs)
513{
514 if (this != &rhs)
515 {
516 if ((this->capacity > 0) && (rhs.count <= this->capacity))
517 {
518 // source array fits into our capacity, copy in place
519 n_assert(0 != this->elements);
520
521 this->CopyRange(this->elements, rhs.elements, rhs.count);
522 if (rhs.count < this->count)
523 {
524 this->DestroyRange(rhs.count, this->count);
525 }
526 this->grow = rhs.grow;
527 this->count = rhs.count;
528 }
529 else
530 {
531 // source array doesn't fit into our capacity, need to reallocate
532 this->Delete();
533 this->Copy(rhs);
534 }
535 }
536}

541template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED> void
542Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::operator=(Array<TYPE, SMALL_VECTOR_SIZE, PINNED>&& rhs) noexcept
543{
544 if (this != &rhs)
545 {
546 this->Delete();
547
548 // If rhs is not using stack, simply reassign pointers
549 if (rhs.elements != rhs.stackElements.data())
550 {
551 this->elements = rhs.elements;
552 rhs.elements = nullptr;
553 }
554 else
555 {
556 // Otherwise, move every element over to the stack of this array
557 this->MoveRange(this->elements, rhs.elements, rhs.count);
558 }
559
560 this->grow = rhs.grow;
561 this->count = rhs.count;
562 this->capacity = rhs.capacity;
563 if constexpr (PINNED)
564 this->maxCommitSize = rhs.maxCommitSize;
565 rhs.count = 0;
566 rhs.capacity = 0;

570//------------------------------------------------------------------------------
573template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED> void
574Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::GrowTo(SizeT newCapacity)
575{
576 if (newCapacity > SMALL_VECTOR_SIZE)
577 {
578 if constexpr (PINNED)
579 {
580 if (this->elements == nullptr)
581 {
582 SizeT pageSize = System::PageSize;
583 SizeT reservationSize = Math::align(this->maxCommitSize * sizeof(TYPE), pageSize);
584 this->elements = (TYPE*)Memory::AllocVirtual(reservationSize);
585 }
586
587 SizeT pageSize = System::PageSize;
588
589 // Total amount of bytes needed to fill new capacity
590 SizeT totalByteSize = newCapacity * sizeof(TYPE);
591
592 // Rounded up to the page size so we don't waste memory we allocate anyways
593 SizeT totalBytesNeeded = Math::align(totalByteSize, pageSize);
594 n_assert(totalBytesNeeded <= this->maxCommitSize);
595 SizeT roundedUpNewCapacity = totalBytesNeeded / sizeof(TYPE);
596 SizeT offset = this->capacity * sizeof(TYPE);
597 if (totalBytesNeeded > offset)
598 {
599 // The amount of bytes we need to commit is the difference between the new and the old capacity
600 SizeT commitSize = (roundedUpNewCapacity - this->capacity) * sizeof(TYPE);
601 this->capacity = roundedUpNewCapacity;
602 Memory::CommitVirtual(((byte*)this->elements) + offset, commitSize);
603 }
604 }
605 else
606 {
607 TYPE* newArray = ArrayAlloc<TYPE>(newCapacity);
608 if (this->elements)
609 {
610 this->MoveRange(newArray, this->elements, this->count);
611
612 // discard old array if not the stack array
613 if (this->elements != this->stackElements.data())
614 ArrayFree(this->capacity, this->elements);
615 }
616 this->elements = newArray;
617 this->capacity = newCapacity;
618 }
619 }
620}

622//------------------------------------------------------------------------------
625template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED>
626void
627Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::Grow()
628{
629 #if NEBULA_BOUNDSCHECKS
630 n_assert(this->grow > 0);
631 #endif
632
633 SizeT growToSize;
634 if (0 == this->capacity)
635 {
636 growToSize = this->grow;
637 }
638 else
639 {
640 // grow by half of the current capacity, but never more then MaxGrowSize
641 SizeT growBy = this->capacity >> 1;
642 if (growBy == 0)
643 {
644 growBy = MinGrowSize;
645 }
646 else if (growBy > MaxGrowSize)
647 {
648 growBy = MaxGrowSize;
649 }
650 growToSize = this->capacity + growBy;
651 }
652 this->GrowTo(growToSize);
653}

660template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED>
661void
662Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::Move(IndexT fromIndex, IndexT toIndex)
663{
664 #if NEBULA_BOUNDSCHECKS
665 n_assert(this->elements);
666 n_assert(fromIndex < this->count);
667 #endif
668
669 // nothing to move?
670 if (fromIndex == toIndex)
671 {
672 return;
673 }
674
675 // compute number of elements to move
676 SizeT num = this->count - fromIndex;
677
678 // check if array needs to grow
679 SizeT neededSize = toIndex + num;
680 while (neededSize > this->capacity)
681 {
682 this->Grow();
683 }
684
685 if (fromIndex > toIndex)
686 {
687 // this is a backward move
688 this->MoveRange(&this->elements[toIndex], &this->elements[fromIndex], num);
689 this->DestroyRange(fromIndex + num - 1, this->count);
690 }
691 else
692 {
693 // this is a forward move
694 int i; // NOTE: this must remain signed for the following loop to work!!!
695 for (i = num - 1; i >= 0; --i)
696 {
697 this->elements[toIndex + i] = this->elements[fromIndex + i];
698 }
699
700 // destroy freed elements
701 this->DestroyRange(fromIndex, toIndex);
702 }
703
704 // adjust array size
705 this->count = toIndex + num;
706}

1499template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED>
1500IndexT
1501Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::BinarySearchIndex(const TYPE& elm) const
1502{
1503 SizeT num = this->Size();
1504 if (num > 0)
1505 {
1506 IndexT half;
1507 IndexT lo = 0;
1508 IndexT hi = num - 1;
1509 IndexT mid;
1510 while (lo <= hi)
1511 {
1512 if (0 != (half = num/2))
1513 {
1514 mid = lo + ((num & 1) ? half : (half - 1));
1515 if (elm < this->elements[mid])
1516 {
1517 hi = mid - 1;
1518 num = num & 1 ? half : half - 1;
1519 }
1520 else if (elm > this->elements[mid])
1521 {
1522 lo = mid + 1;
1523 num = half;
1524 }
1525 else
1526 {
1527 return mid;
1528 }
1529 }
1530 else if (0 != num)
1531 {
1532 if (elm != this->elements[lo])
1533 {
1534 return InvalidIndex;
1535 }
1536 else
1537 {
1538 return lo;
1539 }
1540 }
1541 else
1542 {
1543 break;
1544 }
1545 }
1546 }
1547 return InvalidIndex;
1548}

1645{

1648 // The start offset of the memory has to be aligned to the page size
1649 SizeT numUsedBytes = Math::align(this->count * sizeof(TYPE), System::PageSize);
1650 SizeT numNeededPages = numUsedBytes / System::PageSize;
1651 SizeT numUsedPages = this->capacity * sizeof(TYPE) / System::PageSize;
1652 SizeT numFreeablePages = numUsedPages - numNeededPages;

1662 if (this->elements != this->stackElements.data())
1663 ArrayFree(this->capacity, this->elements);

1785//------------------------------------------------------------------------------
1790template<class TYPE, int SMALL_VECTOR_SIZE, bool PINNED>
1791IndexT
1792Array<TYPE, SMALL_VECTOR_SIZE, PINNED>::InsertSorted(const TYPE& elm)
1793{
1794 SizeT num = this->Size();
1795 if (num == 0)
1796 {
1797 // array is currently empty
1798 this->Append(elm);
1799 return this->Size() - 1;
1800 }
1801 else
1802 {
1803 IndexT half;
1804 IndexT lo = 0;
1805 IndexT hi = num - 1;
1806 IndexT mid;
1807 while (lo <= hi)
1808 {
1809 if (0 != (half = num/2))
1810 {
1811 mid = lo + ((num & 1) ? half : (half - 1));
1812 if (elm < this->elements[mid])
1813 {
1814 hi = mid - 1;
1815 num = num & 1 ? half : half - 1;
1816 }
1817 else if (elm > this->elements[mid])
1818 {
1819 lo = mid + 1;
1820 num = half;
1821 }
1822 else
1823 {
1824 // element already exists at [mid], append the
1825 // new element to the end of the range
1826 return this->InsertAtEndOfIdenticalRange(mid, elm);
1827 }
1828 }
1829 else if (0 != num)
1830 {
1831 if (elm < this->elements[lo])
1832 {
1833 this->Insert(lo, elm);
1834 return lo;
1835 }
1836 else if (elm > this->elements[lo])
1837 {
1838 this->Insert(lo + 1, elm);
1839 return lo + 1;
1840 }
1841 else
1842 {
1843 // element already exists at [low], append
1844 // the new element to the end of the range
1845 return this->InsertAtEndOfIdenticalRange(lo, elm);
1846 }
1847 }
1848 else
1849 {
1850 #if NEBULA_BOUNDSCHECKS
1851 n_assert(0 == lo);
1852 #endif
1853 this->Insert(lo, elm);
1854 return lo;
1855 }
1856 }
1857 if (elm < this->elements[lo])
1858 {
1859 this->Insert(lo, elm);
1860 return lo;
1861 }
1862 else if (elm > this->elements[lo])
1863 {
1864 this->Insert(lo + 1, elm);
1865 return lo + 1;
1866 }
1867 else
1868 {
1869 // can't happen(?)
1870 }
1871 }
1872 // can't happen
1873 n_error("Array::InsertSorted: Can't happen!");
1874 return InvalidIndex;
1875}
