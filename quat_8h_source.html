320inverse(const quat& q)
321{
322 scalar len = lengthsq(q);
323 if (len > 0.00001f)
324 {
325 quat con = conjugate(q);
326 __m128 temp = _mm_set1_ps(1.0f / len);

392rotationquataxis(const vec3& axis, scalar angle)
393{
394 n_assert2(Math::nearequal(length(axis), 1.0f, 0.001f), "axis needs to be normalized");
395

397 float cosangle = Math::cos(0.5f * angle);
398
399 // set w component to 1
400 __m128 b = _mm_and_ps(axis.vec, _mask_xyz);

408__forceinline quat
409rotationquatyawpitchroll(scalar yaw, scalar pitch, scalar roll)
410{
411 scalar halfYaw = 0.5f * yaw;
412 scalar halfPitch = 0.5f * pitch;
413 scalar halfRoll = 0.5f * roll;
414 scalar cosYaw = Math::cos(halfYaw);
415 scalar sinYaw = Math::sin(halfYaw);
416 scalar cosPitch = Math::cos(halfPitch);
417 scalar sinPitch = Math::sin(halfPitch);
418 scalar cosRoll = Math::cos(halfRoll);
419 scalar sinRoll = Math::sin(halfRoll);
420 quat q(-(cosRoll * sinPitch * cosYaw + sinRoll * cosPitch * sinYaw),
421 -(cosRoll * cosPitch * sinYaw - sinRoll * sinPitch * cosYaw),
422 -(sinRoll * cosPitch * cosYaw - cosRoll * sinPitch * sinYaw),
423 -(cosRoll * cosPitch * cosYaw + sinRoll * sinPitch * sinYaw));
424
425
426 return q;
427}

434__forceinline quat
435slerp(const quat& q1, const quat& q2, scalar t)
436{
437 __m128 to;
438
439 float qdot = dot(q1, q2);
440 // flip when negative angle
441 if (qdot < 0)
442 {
443 qdot = -qdot;
444 to = _mm_mul_ps(q2.vec, _mm_set_ps1(-1.0f));
445 }
446 else
447 {
448 to = q2.vec;
449 }
450
451 // just lerp when angle between is narrow
452 if (qdot < 0.95f)
453 {
454 //dont break acos
455 float clamped = Math::clamp(qdot, -1.0f, 1.0f);
456 float angle = Math::acos(clamped);
457
458 float sin_angle = Math::sin(angle);
459 float sin_angle_t = Math::sin(angle * t);
460 float sin_omega_t = Math::sin(angle * (1.0f - t));
461
462 __m128 s0 = _mm_set_ps1(sin_omega_t);
463 __m128 s1 = _mm_set_ps1(sin_angle_t);
464 __m128 sin_div = _mm_set_ps1(1.0f / sin_angle);
465 return _mm_mul_ps(_mm_add_ps(_mm_mul_ps(q1.vec, s0), _mm_mul_ps(to, s1)), sin_div);
466 }
467 else
468 {
469
470 float scale0 = 1.0f - t;
471 float scale1 = t;
472 __m128 s0 = _mm_set_ps1(scale0);
473 __m128 s1 = _mm_set_ps1(scale1);
474
475 return _mm_add_ps(_mm_mul_ps(q1.vec, s0), _mm_mul_ps(to, s1));
476 }
