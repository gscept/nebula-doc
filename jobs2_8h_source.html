169template <typename CTX> void
170JobDispatch(
171 const JobFunc& func
172 , const SizeT numInvocations
173 , const SizeT groupSize
174 , const CTX& context
175 , const Util::FixedArray<const Threading::AtomicCounter*>& waitCounters = nullptr
176 , Threading::AtomicCounter* doneCounter = nullptr
177 , Threading::Event* signalEvent = nullptr
178)
179{
180 static_assert(std::is_trivially_destructible<CTX>::value, "Job context has to be trivially destructible");
181 n_assert(numInvocations > 0);
182 n_assert(doneCounter != nullptr ? *doneCounter > 0 : true);
183
184 // Calculate the number of actual jobs based on invocations and group size
185 SizeT numJobs = Math::ceil(numInvocations / float(groupSize));
186
187 // Calculate allocation size which is node + counters + data context
188 auto dynamicAllocSize = sizeof(JobNode) + sizeof(CTX) + waitCounters.Size() * sizeof(const Threading::AtomicCounter*);
189 auto mem = JobAlloc<char>(dynamicAllocSize);
190 auto node = (JobNode*)mem;
191
192 // Copy over wait counters
193 node->job.waitCounters = nullptr;
194 if (waitCounters.Size() > 0)
195 {
196 node->job.waitCounters = (const Threading::AtomicCounter**)(mem + sizeof(JobNode) + sizeof(CTX));
197 memcpy(node->job.waitCounters, waitCounters.Begin(), waitCounters.Size() * sizeof(const Threading::AtomicCounter*));
198 }
199
200 // Move context
201 node->job.data = (void*)(mem + sizeof(JobNode));
202 auto data = reinterpret_cast<CTX*>(node->job.data);
203 *data = context;
204
205 node->job.func = func;
206 node->job.remainingGroups = numJobs;
207 node->job.groupCompletionCounter = numJobs;
208 node->job.numInvocations = numInvocations;
209 node->job.groupSize = groupSize;
210 node->job.numWaitCounters = (SizeT)waitCounters.Size();
211 node->job.doneCounter = doneCounter;
212 node->job.signalEvent = signalEvent;
213 node->sequence = nullptr;
214
215 // Add to end of linked list
216 ctx.jobLock.Enter();
217
218 // First, set head node if nullptr
219 if (ctx.head == nullptr)
220 ctx.head = node;
221
222 // Then add node to end of list
223 node->next = nullptr;
224 if (ctx.tail != nullptr)
225 ctx.tail->next = node;
226 ctx.tail = node;
227
228 ctx.jobLock.Leave();
229
230 // Trigger threads to wake up and compete for jobs
231 for (Ptr<JobThread>& thread : ctx.threads)
232 {
233 thread->SignalWorkAvailable();
234 }
235}
