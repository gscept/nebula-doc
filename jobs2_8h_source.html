157template <typename CTX> void
158JobDispatch(
159 const JobFunc& func
160 , const SizeT numInvocations
161 , const SizeT groupSize
162 , const CTX& context
163 , const Util::FixedArray<const Threading::AtomicCounter*>& waitCounters = nullptr
164 , Threading::AtomicCounter* doneCounter = nullptr
165 , Threading::Event* signalEvent = nullptr
166)
167{
168 static_assert(std::is_trivially_destructible<CTX>::value, "Job context has to be trivially destructible");
169 n_assert(doneCounter != nullptr ? *doneCounter > 0 : true);
170
171 // Calculate the number of actual jobs based on invocations and group size
172 SizeT numJobs = Math::ceil(numInvocations / float(groupSize));
173
174 // Calculate allocation size which is node + counters + data context
175 auto dynamicAllocSize = sizeof(JobNode) + sizeof(CTX) + waitCounters.Size() * sizeof(const Threading::AtomicCounter*);
176 auto mem = JobAlloc<char>(dynamicAllocSize);
177 auto node = (JobNode*)mem;
178
179 // Copy over wait counters
180 node->job.waitCounters = nullptr;
181 if (waitCounters.Size() > 0)
182 {
183 node->job.waitCounters = (const Threading::AtomicCounter**)(mem + sizeof(JobNode) + sizeof(CTX));
184 memcpy(node->job.waitCounters, waitCounters.Begin(), waitCounters.Size() * sizeof(const Threading::AtomicCounter*));
185 }
186
187 // Move context
188 node->job.data = (void*)(mem + sizeof(JobNode));
189 auto data = reinterpret_cast<CTX*>(node->job.data);
190 *data = context;
191
192 node->job.func = func;
193 node->job.remainingGroups = numJobs;
194 node->job.groupCompletionCounter = numJobs;
195 node->job.numInvocations = numInvocations;
196 node->job.groupSize = groupSize;
197 node->job.numWaitCounters = (SizeT)waitCounters.Size();
198 node->job.doneCounter = doneCounter;
199 node->job.signalEvent = signalEvent;
200 node->sequenceWaitValue = 0;
201
202 // Add to end of linked list
203 ctx.jobLock.Enter();
204
205 // First, set head node if nullptr
206 if (ctx.head == nullptr)
207 ctx.head = node;
208
209 // Then add node to end of list
210 node->next = nullptr;
211 if (ctx.tail != nullptr)
212 ctx.tail->next = node;
213 ctx.tail = node;
214
215 ctx.jobLock.Leave();
216
217 // Trigger threads to wake up and compete for jobs
218 for (Ptr<JobThread>& thread : ctx.threads)
219 {
220 thread->SignalWorkAvailable();
221 }
222}

243template<typename CTX> void
244JobAppendSequence(const JobFunc& func, const SizeT numInvocations, const SizeT groupSize, const CTX& context)
245{
246 static_assert(std::is_trivially_destructible<CTX>::value, "Job context has to be trivially destructible");
247
248 // Calculate the number of actual jobs based on invocations and group size
249 SizeT numJobs = Math::ceil(numInvocations / float(groupSize));
250
251 // Calculate allocation size which is node + counters + data context
252 auto dynamicAllocSize = sizeof(JobNode) + sizeof(CTX);
253 if (sequenceWaitCounters.IsEmpty())
254 dynamicAllocSize += sizeof(Threading::AtomicCounter*);
255 else
256 dynamicAllocSize += sequenceWaitCounters.ByteSize();
257
258 auto mem = JobAlloc<char>(dynamicAllocSize);
259 auto node = (JobNode*)mem;
260
261 // Setup pointer to wait counters
262 node->job.waitCounters = (const Threading::AtomicCounter**)(mem + sizeof(JobNode) + sizeof(CTX));
263
264 // If we have sequence wait counters, add them
265 if (!sequenceWaitCounters.IsEmpty())
266 {
267 node->job.numWaitCounters = sequenceWaitCounters.Size();
268 memcpy(node->job.waitCounters, sequenceWaitCounters.Begin(), sequenceWaitCounters.ByteSize());
269
270 // Then clear them, we only allow wait counters for the first job in the sequence
271 sequenceWaitCounters.Clear();
272 }
273 else
274 {
275 node->job.waitCounters[0] = nullptr;
276 }
277
278 // Move context
279 node->job.data = (void*)(mem + sizeof(JobNode));
280 auto data = reinterpret_cast<CTX*>(node->job.data);
281 *data = context;
282
283 node->job.func = func;
284 node->job.remainingGroups = numJobs;
285 node->job.groupCompletionCounter = numJobs;
286 node->job.numInvocations = numInvocations;
287 node->job.groupSize = groupSize;
288 node->job.numWaitCounters = sequenceWaitCounters.Size();
289 node->job.doneCounter = nullptr;
290 node->job.signalEvent = nullptr;
291 node->next = nullptr;
292
293 // Queue job
294 ctx.queuedJobs.Append(node);
295}
