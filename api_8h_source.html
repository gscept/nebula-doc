377 void SetInclusive(Game::FilterBuilder::FilterCreateInfo& filterInfo, size_t const i)
378 {
379 using UnqualifiedType = typename std::remove_const<typename std::remove_reference<TYPE>::type>::type;
380
381 filterInfo.inclusive[i] = UnqualifiedType::ID();
382 filterInfo.access[i] = std::is_const<typename std::remove_reference<TYPE>::type>() ? Game::AccessMode::READ : Game::AccessMode::WRITE;
383 filterInfo.numInclusive++;

386 template<typename...TYPES, std::size_t...Is>
387 void UnrollInclusiveComponents(Game::FilterBuilder::FilterCreateInfo& filterInfo, std::index_sequence<Is...>)
388 {
389 (SetInclusive<typename std::tuple_element<Is, std::tuple<TYPES...>>::type>(filterInfo, Is), ...);

392 template<typename...TYPES, std::size_t...Is>
393 void update_expander(std::function<void(TYPES...)> const& func, Game::Dataset::EntityTableView const& view, const IndexT instance, std::index_sequence<Is...>)
394 {
395 // this is a terribly unreadable line. Here's what it does:
396 // it unpacks the the index sequence and TYPES into individual parameters for func
397 // because we need to cast void pointers (the view buffers), we need to remove any const and reference qualifiers from the type.
398 func((*(((typename std::remove_const<typename std::remove_reference<TYPES>::type>::type*)view.buffers[Is]) + instance))...);

406template<typename ...TYPES>
407inline Game::ProcessorHandle
408RegisterUpdateFunction(World* world, Util::StringAtom name, std::function<void(TYPES...)> func, std::initializer_list<ComponentId> additionalInclusive, std::initializer_list<ComponentId> exclusive)
409{
410 n_assert(exclusive.size() < 0xFF);
411 n_assert(additionalInclusive.size() < 0xFF);
412
413 ProcessorFrameCallback processor = [func](World* world, Game::Dataset data) {
414 for (int v = 0; v < data.numViews; v++)
415 {
416 Game::Dataset::EntityTableView const& view = data.views[v];
417
418 for (IndexT i = 0; i < view.numInstances; ++i)
419 {
420 Internal::update_expander<TYPES...>(func, view, i, std::make_index_sequence<sizeof...(TYPES)>());
421 }
422 }
423 };
424
425 Game::FilterBuilder::FilterCreateInfo filterInfo;
426 Internal::UnrollInclusiveComponents<TYPES...>(filterInfo, std::make_index_sequence<sizeof...(TYPES)>());
427 for (int i = 0; i < (int)additionalInclusive.size(); i++)
428 filterInfo.inclusive[filterInfo.numInclusive + i] = *(additionalInclusive.begin() + i);
429 filterInfo.numInclusive = filterInfo.numInclusive + (uint8_t)additionalInclusive.size();
430 for (int i = 0; i < exclusive.size(); i++)
431 filterInfo.exclusive[i] = *(exclusive.begin() + i);
432 filterInfo.numExclusive = (uint8_t)exclusive.size();
433 Game::Filter filter = Game::FilterBuilder::CreateFilter(filterInfo);
434 Game::ProcessorCreateInfo processorInfo;
435 processorInfo.async = false;
436 processorInfo.filter = filter;
437 processorInfo.name = name;
438 processorInfo.OnBeginFrame = processor;
439
440 Game::ProcessorHandle pHandle = Game::CreateProcessor(processorInfo);
441 return pHandle;
442}
