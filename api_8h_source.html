411 template<class TYPE>
412 void SetInclusive(Game::FilterCreateInfo& filterInfo, size_t const i)
413 {
414 using UnqualifiedType = typename std::remove_const<typename std::remove_reference<TYPE>::type>::type;
415
416 filterInfo.inclusive[i] = UnqualifiedType::ID();
417 filterInfo.access[i] = std::is_const<typename std::remove_reference<TYPE>::type>() ? Game::AccessMode::READ : Game::AccessMode::WRITE;
418 filterInfo.numInclusive++;
419 }

421 template<typename...TYPES, std::size_t...Is>
422 void UnrollInclusiveProperties(Game::FilterCreateInfo& filterInfo, std::index_sequence<Is...>)
423 {
424 (SetInclusive<typename std::tuple_element<Is, std::tuple<TYPES...>>::type>(filterInfo, Is), ...);
425 }

427 template<typename...TYPES, std::size_t...Is>
428 void update_expander(std::function<void(TYPES...)> const& func, Game::Dataset::CategoryTableView const& view, const IndexT instance, std::index_sequence<Is...>)
429 {
430 // this is a terribly unreadable line. Here's what it does:
431 // it unpacks the the index sequence and TYPES into individual parameters for func
432 // because we need to cast void pointers (the view buffers), we need to remove any const and reference qualifiers from the type.
433 func((*(((typename std::remove_const<typename std::remove_reference<TYPES>::type>::type*)view.buffers[Is]) + instance))...);
434 }

441template<typename ...TYPES>
442inline Game::ProcessorHandle
443RegisterUpdateFunction(World* world, Util::StringAtom name, std::function<void(TYPES...)> func, std::initializer_list<PropertyId> additionalInclusive, std::initializer_list<PropertyId> exclusive)
444{
445 n_assert(exclusive.size() < 0xFF);
446 n_assert(additionalInclusive.size() < 0xFF);
447
448 ProcessorFrameCallback processor = [func](World* world, Game::Dataset data) {
449 for (int v = 0; v < data.numViews; v++)
450 {
451 Game::Dataset::CategoryTableView const& view = data.views[v];
452
453 for (IndexT i = 0; i < view.numInstances; ++i)
454 {
455 Internal::update_expander<TYPES...>(func, view, i, std::make_index_sequence<sizeof...(TYPES)>());
456 }
457 }
458 };
459
460 Game::FilterCreateInfo filterInfo;
461 Internal::UnrollInclusiveProperties<TYPES...>(filterInfo, std::make_index_sequence<sizeof...(TYPES)>());
462 for (int i = 0; i < (int)additionalInclusive.size(); i++)
463 filterInfo.inclusive[filterInfo.numInclusive + i] = *(additionalInclusive.begin() + i);
464 filterInfo.numInclusive = filterInfo.numInclusive + (uint8_t)additionalInclusive.size();
465 for (int i = 0; i < exclusive.size(); i++)
466 filterInfo.exclusive[i] = *(exclusive.begin() + i);
467 filterInfo.numExclusive = (uint8_t)exclusive.size();
468 Game::Filter filter = Game::CreateFilter(filterInfo);
469 Game::ProcessorCreateInfo processorInfo;
470 processorInfo.async = false;
471 processorInfo.filter = filter;
472 processorInfo.name = name;
473 processorInfo.OnBeginFrame = processor;
474
475 Game::ProcessorHandle pHandle = Game::CreateProcessor(processorInfo);
476 return pHandle;
477}
