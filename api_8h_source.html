404 void SetInclusive(Game::FilterCreateInfo& filterInfo, size_t const i)
405 {
406 using UnqualifiedType = typename std::remove_const<typename std::remove_reference<TYPE>::type>::type;
407
408 filterInfo.inclusive[i] = UnqualifiedType::ID();
409 filterInfo.access[i] = std::is_const<typename std::remove_reference<TYPE>::type>() ? Game::AccessMode::READ : Game::AccessMode::WRITE;
410 filterInfo.numInclusive++;
411 }

413 template<typename...TYPES, std::size_t...Is>
414 void UnrollInclusiveProperties(Game::FilterCreateInfo& filterInfo, std::index_sequence<Is...>)
415 {
416 (SetInclusive<typename std::tuple_element<Is, std::tuple<TYPES...>>::type>(filterInfo, Is), ...);
417 }

419 template<typename...TYPES, std::size_t...Is>
420 void update_expander(std::function<void(TYPES...)> const& func, Game::Dataset::CategoryTableView const& view, const IndexT instance, std::index_sequence<Is...>)
421 {
422 // this is a terribly unreadable line. Here's what it does:
423 // it unpacks the the index sequence and TYPES into individual parameters for func
424 // because we need to cast void pointers (the view buffers), we need to remove any const and reference qualifiers from the type.
425 func((*(((typename std::remove_const<typename std::remove_reference<TYPES>::type>::type*)view.buffers[Is]) + instance))...);
426 }

433template<typename ...TYPES>
434inline Game::ProcessorHandle
435RegisterUpdateFunction(World* world, Util::StringAtom name, std::function<void(TYPES...)> func, std::initializer_list<PropertyId> additionalInclusive, std::initializer_list<PropertyId> exclusive)
436{
437 n_assert(exclusive.size() < 0xFF);
438 n_assert(additionalInclusive.size() < 0xFF);
439
440 ProcessorFrameCallback processor = [func](World* world, Game::Dataset data) {
441 for (int v = 0; v < data.numViews; v++)
442 {
443 Game::Dataset::CategoryTableView const& view = data.views[v];
444
445 for (IndexT i = 0; i < view.numInstances; ++i)
446 {
447 Internal::update_expander<TYPES...>(func, view, i, std::make_index_sequence<sizeof...(TYPES)>());
448 }
449 }
450 };
451
452 Game::FilterCreateInfo filterInfo;
453 Internal::UnrollInclusiveProperties<TYPES...>(filterInfo, std::make_index_sequence<sizeof...(TYPES)>());
454 for (int i = 0; i < (int)additionalInclusive.size(); i++)
455 filterInfo.inclusive[filterInfo.numInclusive + i] = *(additionalInclusive.begin() + i);
456 filterInfo.numInclusive = filterInfo.numInclusive + (uint8_t)additionalInclusive.size();
457 for (int i = 0; i < exclusive.size(); i++)
458 filterInfo.exclusive[i] = *(exclusive.begin() + i);
459 filterInfo.numExclusive = (uint8_t)exclusive.size();
460 Game::Filter filter = Game::CreateFilter(filterInfo);
461 Game::ProcessorCreateInfo processorInfo;
462 processorInfo.async = false;
463 processorInfo.filter = filter;
464 processorInfo.name = name;
465 processorInfo.OnBeginFrame = processor;
466
467 Game::ProcessorHandle pHandle = Game::CreateProcessor(processorInfo);
468 return pHandle;
469}
