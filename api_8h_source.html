440 template<class TYPE>
441 void SetInclusive(Game::FilterCreateInfo& filterInfo, size_t const i)
442 {
443 using UnqualifiedType = typename std::remove_const<typename std::remove_reference<TYPE>::type>::type;
444
445 filterInfo.inclusive[i] = UnqualifiedType::ID();
446 filterInfo.access[i] = std::is_const<typename std::remove_reference<TYPE>::type>() ? Game::AccessMode::READ : Game::AccessMode::WRITE;
447 filterInfo.numInclusive++;
448 }

450 template<typename...TYPES, std::size_t...Is>
451 void UnrollInclusiveComponents(Game::FilterCreateInfo& filterInfo, std::index_sequence<Is...>)
452 {
453 (SetInclusive<typename std::tuple_element<Is, std::tuple<TYPES...>>::type>(filterInfo, Is), ...);
454 }
455
456 template<typename...TYPES, std::size_t...Is>
457 void update_expander(std::function<void(TYPES...)> const& func, Game::Dataset::EntityTableView const& view, const IndexT instance, std::index_sequence<Is...>)
458 {
459 // this is a terribly unreadable line. Here's what it does:
460 // it unpacks the the index sequence and TYPES into individual parameters for func
461 // because we need to cast void pointers (the view buffers), we need to remove any const and reference qualifiers from the type.
462 func((*(((typename std::remove_const<typename std::remove_reference<TYPES>::type>::type*)view.buffers[Is]) + instance))...);
463 }

470template<typename ...TYPES>
471inline Game::ProcessorHandle
472RegisterUpdateFunction(World* world, Util::StringAtom name, std::function<void(TYPES...)> func, std::initializer_list<ComponentId> additionalInclusive, std::initializer_list<ComponentId> exclusive)
473{
474 n_assert(exclusive.size() < 0xFF);
475 n_assert(additionalInclusive.size() < 0xFF);
476
477 ProcessorFrameCallback processor = [func](World* world, Game::Dataset data) {
478 for (int v = 0; v < data.numViews; v++)
479 {
480 Game::Dataset::EntityTableView const& view = data.views[v];
481
482 for (IndexT i = 0; i < view.numInstances; ++i)
483 {
484 Internal::update_expander<TYPES...>(func, view, i, std::make_index_sequence<sizeof...(TYPES)>());
485 }
486 }
487 };
488
489 Game::FilterCreateInfo filterInfo;
490 Internal::UnrollInclusiveComponents<TYPES...>(filterInfo, std::make_index_sequence<sizeof...(TYPES)>());
491 for (int i = 0; i < (int)additionalInclusive.size(); i++)
492 filterInfo.inclusive[filterInfo.numInclusive + i] = *(additionalInclusive.begin() + i);
493 filterInfo.numInclusive = filterInfo.numInclusive + (uint8_t)additionalInclusive.size();
494 for (int i = 0; i < exclusive.size(); i++)
495 filterInfo.exclusive[i] = *(exclusive.begin() + i);
496 filterInfo.numExclusive = (uint8_t)exclusive.size();
497 Game::Filter filter = Game::CreateFilter(filterInfo);
498 Game::ProcessorCreateInfo processorInfo;
499 processorInfo.async = false;
500 processorInfo.filter = filter;
501 processorInfo.name = name;
502 processorInfo.OnBeginFrame = processor;
503
504 Game::ProcessorHandle pHandle = Game::CreateProcessor(processorInfo);
505 return pHandle;
506}
