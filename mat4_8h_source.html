368__forceinline scalar
369determinant(const mat4& m)
370{
371 __m128 Va,Vb,Vc;
372 __m128 r1,r2,r3,tt,tt2;
373 __m128 sum,Det;
374
375 __m128 _L1 = m.r[0].vec;
376 __m128 _L2 = m.r[1].vec;
377 __m128 _L3 = m.r[2].vec;
378 __m128 _L4 = m.r[3].vec;
379 // Calculating the minterms for the first line.
380
381 // _mm_ror_ps is just a macro using _mm_shuffle_ps().
382 tt = _L4; tt2 = mm_ror_ps(_L3,1);
383 Vc = _mm_mul_ps(tt2, mm_ror_ps(tt,0)); // V3' dot V4
384 Va = _mm_mul_ps(tt2, mm_ror_ps(tt,2)); // V3' dot V4"
385 Vb = _mm_mul_ps(tt2, mm_ror_ps(tt,3)); // V3' dot V4^
386
387 r1 = _mm_sub_ps(mm_ror_ps(Va,1), mm_ror_ps(Vc,2)); // V3" dot V4^ - V3^ dot V4"
388 r2 = _mm_sub_ps(mm_ror_ps(Vb,2), mm_ror_ps(Vb,0)); // V3^ dot V4' - V3' dot V4^
389 r3 = _mm_sub_ps(mm_ror_ps(Va,0), mm_ror_ps(Vc,1)); // V3' dot V4" - V3" dot V4'
390
391 tt = _L2;
392 Va = mm_ror_ps(tt,1); sum = _mm_mul_ps(Va,r1);
393 Vb = mm_ror_ps(tt,2); sum = _mm_add_ps(sum,_mm_mul_ps(Vb,r2));
394 Vc = mm_ror_ps(tt,3); sum = _mm_add_ps(sum,_mm_mul_ps(Vc,r3));
395
396 // Calculating the determinant.
397 Det = _mm_mul_ps(sum,_L1);
398 Det = _mm_add_ps(Det,_mm_movehl_ps(Det,Det));
399
400 // Calculating the minterms of the second line (using previous results).
401 tt = mm_ror_ps(_L1,1); sum = _mm_mul_ps(tt,r1);
402 tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r2));
403 tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r3));
404
405 // Testing the determinant.
406 Det = _mm_sub_ss(Det,_mm_shuffle_ps(Det,Det,1));
407 return vec4(Det).x;
408}

413__forceinline mat4
414inverse(const mat4& m)
415{
416 __m128 Va,Vb,Vc;
417 __m128 r1,r2,r3,tt,tt2;
418 __m128 sum,Det,RDet;
419 __m128 trns0,trns1,trns2,trns3;
420
421 const __m128i pnpn = _mm_setr_epi32(0x00000000, static_cast<int>(0x80000000), 0x00000000, static_cast<int>(0x80000000));
422 const __m128i npnp = _mm_setr_epi32(static_cast<int>(0x80000000), 0x00000000, static_cast<int>(0x80000000), 0x00000000);
423 const __m128 zeroone = _mm_setr_ps(1.0f, 0.0f, 0.0f, 1.0f);
424
425 __m128 _L1 = m.r[0].vec;
426 __m128 _L2 = m.r[1].vec;
427 __m128 _L3 = m.r[2].vec;
428 __m128 _L4 = m.r[3].vec;
429 // Calculating the minterms for the first line.
430
431 // _mm_ror_ps is just a macro using _mm_shuffle_ps().
432 tt = _L4; tt2 = mm_ror_ps(_L3,1);
433 Vc = _mm_mul_ps(tt2, mm_ror_ps(tt,0)); // V3'dot V4
434 Va = _mm_mul_ps(tt2, mm_ror_ps(tt,2)); // V3'dot V4"
435 Vb = _mm_mul_ps(tt2, mm_ror_ps(tt,3)); // V3' dot V4^
436
437 r1 = _mm_sub_ps(mm_ror_ps(Va,1), mm_ror_ps(Vc,2)); // V3" dot V4^ - V3^ dot V4"
438 r2 = _mm_sub_ps(mm_ror_ps(Vb,2), mm_ror_ps(Vb,0)); // V3^ dot V4' - V3' dot V4^
439 r3 = _mm_sub_ps(mm_ror_ps(Va,0), mm_ror_ps(Vc,1)); // V3' dot V4" - V3" dot V4'
440
441 tt = _L2;
442 Va = mm_ror_ps(tt,1); sum = _mm_mul_ps(Va,r1);
443 Vb = mm_ror_ps(tt,2); sum = _mm_add_ps(sum,_mm_mul_ps(Vb,r2));
444 Vc = mm_ror_ps(tt,3); sum = _mm_add_ps(sum,_mm_mul_ps(Vc,r3));
445
446 // Calculating the determinant.
447 Det = _mm_mul_ps(sum,_L1);
448 Det = _mm_add_ps(Det,_mm_movehl_ps(Det,Det));
449
450
451 __m128 mtL1 = _mm_xor_ps(sum, _mm_castsi128_ps(pnpn));
452
453 // Calculating the minterms of the second line (using previous results).
454 tt = mm_ror_ps(_L1,1); sum = _mm_mul_ps(tt,r1);
455 tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r2));
456 tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r3));
457 __m128 mtL2 = _mm_xor_ps(sum, _mm_castsi128_ps(npnp));
458
459 // Testing the determinant.
460 Det = _mm_sub_ss(Det,_mm_shuffle_ps(Det,Det,1));
461
462 // Calculating the minterms of the third line.
463 tt = mm_ror_ps(_L1,1);
464 Va = _mm_mul_ps(tt,Vb); // V1' dot V2"
465 Vb = _mm_mul_ps(tt,Vc); // V1' dot V2^
466 Vc = _mm_mul_ps(tt,_L2); // V1' dot V2
467
468 r1 = _mm_sub_ps(mm_ror_ps(Va,1), mm_ror_ps(Vc,2)); // V1" dot V2^ - V1^ dot V2"
469 r2 = _mm_sub_ps(mm_ror_ps(Vb,2), mm_ror_ps(Vb,0)); // V1^ dot V2' - V1' dot V2^
470 r3 = _mm_sub_ps(mm_ror_ps(Va,0), mm_ror_ps(Vc,1)); // V1' dot V2" - V1" dot V2'
471
472 tt = mm_ror_ps(_L4,1); sum = _mm_mul_ps(tt,r1);
473 tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r2));
474 tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r3));
475 __m128 mtL3 = _mm_xor_ps(sum, _mm_castsi128_ps(pnpn));
476
477 // Dividing is FASTER than rcp_nr! (Because rcp_nr causes many register-memory RWs).
478 RDet = _mm_div_ss(zeroone, Det); // TODO: just 1.0f?
479 RDet = _mm_shuffle_ps(RDet,RDet,0x00);
480
481 // Devide the first 12 minterms with the determinant.
482 mtL1 = _mm_mul_ps(mtL1, RDet);
483 mtL2 = _mm_mul_ps(mtL2, RDet);
484 mtL3 = _mm_mul_ps(mtL3, RDet);
485
486 // Calculate the minterms of the forth line and devide by the determinant.
487 tt = mm_ror_ps(_L3,1); sum = _mm_mul_ps(tt,r1);
488 tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r2));
489 tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r3));
490 __m128 mtL4 = _mm_xor_ps(sum, _mm_castsi128_ps(npnp));
491 mtL4 = _mm_mul_ps(mtL4, RDet);
492
493 // Now we just have to transpose the minterms matrix.
494 trns0 = _mm_unpacklo_ps(mtL1,mtL2);
495 trns1 = _mm_unpacklo_ps(mtL3,mtL4);
496 trns2 = _mm_unpackhi_ps(mtL1,mtL2);
497 trns3 = _mm_unpackhi_ps(mtL3,mtL4);
498 _L1 = _mm_movelh_ps(trns0,trns1);
499 _L2 = _mm_movehl_ps(trns1,trns0);
500 _L3 = _mm_movelh_ps(trns2,trns3);
501 _L4 = _mm_movehl_ps(trns3,trns2);
502
503 return mat4(vec4(_L1), vec4(_L2), vec4(_L3), vec4(_L4));
504}

510lookatlh(const point& eye, const point& at, const vector& up)
511{
512#if NEBULA_DEBUG
513 n_assert(length(up) > 0);
514#endif
515 // hmm the XM lookat functions are kinda pointless, because they
516 // return a VIEW matrix, which is already inverse (so one would
517 // need to reverse again!)
518 const vector zaxis = normalize(at - eye);
519 vector normUp = normalize(up);
520 if (Math::abs(dot(zaxis, normUp)) > 0.9999999f)
521 {
522 // need to choose a different up vector because up and lookat point
523 // into same or opposite direction
524 // just rotate y->x, x->z and z->y
525 normUp = permute(normUp, normUp, 1, 2, 0);
526 }
527 const vector xaxis = normalize(cross(normUp, zaxis));
528 const vector yaxis = normalize(cross(zaxis, xaxis));
529 return mat4(xaxis, yaxis, zaxis, eye);
530}

560static inline __m256 twolincomb_AVX_8(__m256 A01, const mat4 &B)
561{
562 __m256 result;
563 result = _mm256_mul_ps(_mm256_shuffle_ps(A01, A01, 0x00), _mm256_broadcast_ps(&B.r[0].vec));
564 result = _mm256_add_ps(result, _mm256_mul_ps(_mm256_shuffle_ps(A01, A01, 0x55), _mm256_broadcast_ps(&B.r[1].vec)));

566 result = _mm256_add_ps(result, _mm256_mul_ps(_mm256_shuffle_ps(A01, A01, 0xff), _mm256_broadcast_ps(&B.r[3].vec)));
567 return result;

933__forceinline mat4
934rotationaxis(const vec3& axis, scalar angle)
935{
936 __m128 norm = normalize(axis).vec;
937
938 scalar sangle = Math::sin(angle);
939 scalar cangle = Math::cos(angle);
940
941 __m128 m1_c = _mm_set_ps1(1.0f - cangle);
942 __m128 c = _mm_set_ps1(cangle);
943 __m128 s = _mm_set_ps1(sangle);
944
945 __m128 nn1 = _mm_shuffle_ps(norm,norm,_MM_SHUFFLE(3,0,2,1));
946 __m128 nn2 = _mm_shuffle_ps(norm,norm,_MM_SHUFFLE(3,1,0,2));
947
948 __m128 v = _mm_mul_ps(nn1,m1_c);
949 v = _mm_mul_ps(nn2,v);
950
951 __m128 nn3 = _mm_mul_ps(norm, m1_c);
952 nn3 = _mm_mul_ps(norm, nn3);
953 nn3 = _mm_add_ps(nn3, c);
954
955 __m128 nn4 = _mm_mul_ps(norm,s);
956 nn4 = _mm_add_ps(nn4, v);
957 __m128 nn5 = _mm_mul_ps(s, norm);
958 nn5 = _mm_sub_ps(v,nn5);
959
960 v = _mm_and_ps(nn3, _mask_xyz);
961
962 __m128 v1 = _mm_shuffle_ps(nn4,nn5,_MM_SHUFFLE(2,1,2,0));
963 v1 = _mm_shuffle_ps(v1,v1,_MM_SHUFFLE(0,3,2,1));
964
965 __m128 v2 = _mm_shuffle_ps(nn4,nn5,_MM_SHUFFLE(0,0,1,1));
966 v2 = _mm_shuffle_ps(v2,v2,_MM_SHUFFLE(2,0,2,0));
967
968
969 nn5 = _mm_shuffle_ps(v,v1,_MM_SHUFFLE(1,0,3,0));
970 nn5 = _mm_shuffle_ps(nn5,nn5,_MM_SHUFFLE(1,3,2,0));
971
972 mat4 m;
973 m.row0 = nn5;
974
975 nn5 = _mm_shuffle_ps(v,v1,_MM_SHUFFLE(3,2,3,1));
976 nn5 = _mm_shuffle_ps(nn5,nn5,_MM_SHUFFLE(1,3,0,2));
977 m.row1 = nn5;
978
979 v2 = _mm_shuffle_ps(v2,v,_MM_SHUFFLE(3,2,1,0));
980 m.row2 = v2;
981
982 m.row3 = _id_w;
