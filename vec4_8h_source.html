532__forceinline vec4
533catmullrom(const vec4& v0, const vec4& v1, const vec4& v2, const vec4& v3, scalar s)
534{
535 scalar s2 = s * s;
536 scalar s3 = s * s2;
537
538 __m128 P0 = _mm_set_ps1((-s3 + 2.0f * s2 - s) * 0.5f);
539 __m128 P1 = _mm_set_ps1((3.0f * s3 - 5.0f * s2 + 2.0f) * 0.5f);
540 __m128 P2 = _mm_set_ps1((-3.0f * s3 + 4.0f * s2 + s) * 0.5f);
541 __m128 P3 = _mm_set_ps1((s3 - s2) * 0.5f);
542
543 P0 = _mm_mul_ps(P0, v0.vec);
544 P1 = _mm_mul_ps(P1, v1.vec);
545 P2 = _mm_mul_ps(P2, v2.vec);
546 P3 = _mm_mul_ps(P3, v3.vec);
547 P0 = _mm_add_ps(P0, P1);
548 P2 = _mm_add_ps(P2, P3);
549 P0 = _mm_add_ps(P0, P2);
550 return P0;

556__forceinline vec4
557hermite(const vec4& v1, const vec4& t1, const vec4& v2, const vec4& t2, scalar s)
558{
559 scalar s2 = s * s;
560 scalar s3 = s * s2;
561
562 __m128 P0 = _mm_set_ps1(2.0f * s3 - 3.0f * s2 + 1.0f);
563 __m128 T0 = _mm_set_ps1(s3 - 2.0f * s2 + s);
564 __m128 P1 = _mm_set_ps1(-2.0f * s3 + 3.0f * s2);
565 __m128 T1 = _mm_set_ps1(s3 - s2);
566
567 __m128 vResult = _mm_mul_ps(P0, v1.vec);
568 __m128 vTemp = _mm_mul_ps(T0, t1.vec);
569 vResult = _mm_add_ps(vResult, vTemp);
570 vTemp = _mm_mul_ps(P1, v2.vec);
571 vResult = _mm_add_ps(vResult, vTemp);
572 vTemp = _mm_mul_ps(T1, t2.vec);
573 vResult = _mm_add_ps(vResult, vTemp);
574 return vResult;

580__forceinline scalar
581angle(const vec4& v0, const vec4& v1)
582{
583
584 __m128 l0 = _mm_mul_ps(v0.vec, v0.vec);
585 l0 = _mm_add_ps(_mm_shuffle_ps(l0, l0, _MM_SHUFFLE(0, 0, 0, 0)),
586 _mm_add_ps(_mm_shuffle_ps(l0, l0, _MM_SHUFFLE(1, 1, 1, 1)), _mm_shuffle_ps(l0, l0, _MM_SHUFFLE(2, 2, 2, 2))));
587
588 __m128 l1 = _mm_mul_ps(v1.vec, v1.vec);
589 l1 = _mm_add_ps(_mm_shuffle_ps(l1, l1, _MM_SHUFFLE(0, 0, 0, 0)),
590 _mm_add_ps(_mm_shuffle_ps(l1, l1, _MM_SHUFFLE(1, 1, 1, 1)), _mm_shuffle_ps(l1, l1, _MM_SHUFFLE(2, 2, 2, 2))));
591
592 __m128 l = _mm_shuffle_ps(l0, l1, _MM_SHUFFLE(0, 0, 0, 0));
593 l = _mm_rsqrt_ps(l);
594 l = _mm_mul_ss(_mm_shuffle_ps(l, l, _MM_SHUFFLE(0, 0, 0, 0)), _mm_shuffle_ps(l, l, _MM_SHUFFLE(1, 1, 1, 1)));
595
596
597 __m128 dot = _mm_mul_ps(v0.vec, v1.vec);
598 dot = _mm_add_ps(_mm_shuffle_ps(dot, dot, _MM_SHUFFLE(0, 0, 0, 0)),
599 _mm_add_ps(_mm_shuffle_ps(dot, dot, _MM_SHUFFLE(1, 1, 1, 1)),
600 _mm_add_ps(_mm_shuffle_ps(dot, dot, _MM_SHUFFLE(2, 2, 2, 2)), _mm_shuffle_ps(dot, dot, _MM_SHUFFLE(3, 3, 3, 3)))));
601
602 dot = _mm_mul_ss(dot, l);
603
604 dot = _mm_max_ss(dot, _minus1);
605 dot = _mm_min_ss(dot, _plus1);
606
607 scalar cangle;
608 _mm_store_ss(&cangle, dot);
609 return Math::acos(cangle);
610}

1039__forceinline vec4
1040permute(const vec4& v0, const vec4& v1, unsigned int i0, unsigned int i1, unsigned int i2, unsigned int i3)
1041{
1042 static __m128i three = _mm_set_epi32(3, 3, 3, 3);
1043
1044 NEBULA_ALIGN16 unsigned int elem[4] = { i0, i1, i2, i3 };
1045 __m128i vControl = _mm_load_si128(reinterpret_cast<const __m128i*>(&elem[0]));
1046
1047 __m128i vSelect = _mm_cmpgt_epi32(vControl, three);
1048 vControl = _mm_and_si128(vControl, three);
1049
1050 __m128 shuffled1 = _mm_permutevar_ps(v0.vec, vControl);
1051 __m128 shuffled2 = _mm_permutevar_ps(v1.vec, vControl);
1052
1053 __m128 masked1 = _mm_andnot_ps(_mm_castsi128_ps(vSelect), shuffled1);
1054 __m128 masked2 = _mm_and_ps(_mm_castsi128_ps(vSelect), shuffled2);
1055
1056 return _mm_or_ps(masked1, masked2);
1057}
