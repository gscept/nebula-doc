248template <typename COMPONENT_TYPE>
249ComponentId
250World::RegisterType(ComponentRegisterInfo<COMPONENT_TYPE> info)
251{
252 uint32_t componentFlags = 0;
253 componentFlags |= (uint32_t)COMPONENTFLAG_DECAY * (uint32_t)info.decay;
254
255 ComponentInterface* cInterface = new ComponentInterface(
256 COMPONENT_TYPE::Traits::name,
257 COMPONENT_TYPE(),
258 componentFlags
259 );
260 cInterface->Init = reinterpret_cast<ComponentInterface::ComponentInitFunc>(info.OnInit);
261 Game::ComponentId const cid = MemDb::AttributeRegistry::Register<COMPONENT_TYPE>(cInterface);
262 Game::ComponentSerialization::Register<COMPONENT_TYPE>(cid);
263 Game::ComponentInspection::Register(cid, &Game::ComponentDrawFuncT<COMPONENT_TYPE>);
264 return cid;
265}

272World::SetComponent(Entity entity, TYPE value)
273{
274#if NEBULA_DEBUG
275 n_assert2(
276 !this->pipeline.IsRunningAsync(),
277 "Settings components in arbitrary entities while executing an async processor is currently not supported!"
278 );
279#endif
280
281#if NEBULA_DEBUG
282 n_assert2(
283 sizeof(TYPE) == MemDb::AttributeRegistry::TypeSize(Game::GetComponentId<TYPE>()),
284 "SetComponent: Provided value's type is not the correct size for the given ComponentId."
285 );
286#endif
287 EntityMapping mapping = this->GetEntityMapping(entity);
288 TYPE* ptr = (TYPE*)this->GetInstanceBuffer(mapping.table, mapping.instance.partition, GetComponentId<TYPE>());
289 *(ptr + mapping.instance.index) = value;

349template <typename TYPE>
350inline TYPE*
351World::AddComponent(Entity entity)
352{
353 /*
354 We could possibly support this by either having thread local allocators and
355 staged queues and gathering after the processors have finished (must clear
356 allocators by issuing a separate job for this at the end of the frame), or by
357 just introducing a simple mutex.
358 */
359#if NEBULA_DEBUG
360 n_assert2(!this->pipeline.IsRunningAsync(), "Adding component to entities while in an async processor is currently not supported!");
361#endif
362 Game::ComponentId id = Game::GetComponentId<TYPE>();
363#if _DEBUG
364 n_assert(MemDb::AttributeRegistry::TypeSize(id) == sizeof(TYPE));
365#endif
366 TYPE* data = this->componentStageAllocator.Alloc<TYPE>();
367 const void* defaultValue = MemDb::AttributeRegistry::DefaultValue(id);
368 Memory::Copy(defaultValue, data, sizeof(TYPE));
369
370 AddStagedComponentCommand cmd = {
371 .entity = entity,
372 .componentId = id,
373 .dataSize = sizeof(TYPE),
374 .data = data,
375 };
376 this->addStagedQueue.Append(cmd);
377
378 MemDb::Attribute* attr = MemDb::AttributeRegistry::GetAttribute(id);
379 ComponentInterface* cInterface;
380 cInterface = static_cast<ComponentInterface*>(attr);
381
382 if (cInterface->Init != nullptr)
383 {
384 // run initialization function if it exists.
385 cInterface->Init(this, entity, data);
386 }
387
388 return data;
389}
