221inline void
222GraphicsContext::InternalDefragment(ID_ALLOCATOR& allocator, Graphics::GraphicsContextState&& state)
223{
224 auto& freeIds = allocator.FreeIds();
225 uint32_t index;
226 uint32_t oldIndex;
227 Graphics::GraphicsEntityId lastId;
228 IndexT mapIndex;
229 uint32_t dataSize;
230 SizeT size = freeIds.Size();
231 for (SizeT i = size - 1; i >= 0; --i)
232 {
233 index = freeIds.Back();
234 freeIds.EraseBack();
235 dataSize = (uint32_t)allocator.Size();
236 if (index >= dataSize)
237 {
238 continue;
239 }
240 oldIndex = dataSize - 1;
241 lastId = state.entities[oldIndex].id;
242 if (state.OnInstanceMoved != nullptr)
243 state.OnInstanceMoved(index, oldIndex);
244 allocator.EraseIndexSwap(index);
245 state.entities.EraseIndexSwap(index);
246 mapIndex = state.entitySliceMap.FindIndex(lastId);
247 if (mapIndex != InvalidIndex)
248 {
249 state.entitySliceMap.ValueAtIndex(lastId, mapIndex) = index;
250 }
251 else
252 {
253 freeIds.Append(index);
254 i++;
255 }
256 }
257 freeIds.Clear();
258}
