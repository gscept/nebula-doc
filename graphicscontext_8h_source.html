185inline void
186GraphicsContext::InternalDefragment(ID_ALLOCATOR& allocator, Graphics::GraphicsContextState&& state)
187{
188 auto& freeIds = allocator.FreeIds();
189 uint32_t index;
190 uint32_t oldIndex;
191 Graphics::GraphicsEntityId lastId;
192 IndexT mapIndex;
193 uint32_t dataSize;
194 SizeT size = freeIds.Size();
195 for (SizeT i = size - 1; i >= 0; --i)
196 {
197 index = freeIds.Back();
198 freeIds.EraseBack();
199 dataSize = (uint32_t)allocator.Size();
200 if (index >= dataSize)
201 {
202 continue;
203 }
204 oldIndex = dataSize - 1;
205 lastId = state.entities[oldIndex].id;
206 if (state.OnInstanceMoved != nullptr)
207 state.OnInstanceMoved(index, oldIndex);
208 allocator.EraseIndexSwap(index);
209 state.entities.EraseIndexSwap(index);
210 mapIndex = state.entitySliceMap.FindIndex(lastId);
211 if (mapIndex != InvalidIndex)
212 {
213 state.entitySliceMap.ValueAtIndex(lastId, mapIndex) = index;
214 }
215 else
216 {
217 freeIds.Append(index);
218 i++;
219 }
220 }
221 freeIds.Clear();
222}
