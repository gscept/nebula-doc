153LockFreeQueue<TYPE>::Resize(const SizeT size)
154{
155 this->capacity = size;
156 this->storage.Resize(size);

158 // setup free list
159 Node* prev = nullptr;
160 for (IndexT i = 0; i < this->storage.Size(); i++)
161 {
162 if (prev != nullptr)
163 prev->next = &this->storage[i];

166 this->freeHead = this->storage.Begin();

170//------------------------------------------------------------------------------
173template<class TYPE>
174inline void
175LockFreeQueue<TYPE>::Enqueue(const TYPE& item)
176{
177 // increase size
178 int size = Interlocked::Increment(&this->size);
179 n_assert(size <= this->capacity);
180
181 // allocate a new node from storage
182 Node* newNode = this->Alloc();
183 newNode->data = item;
184 newNode->next = nullptr;
185
186 Node* oldTail = nullptr;
187 while (true)
188 {
189 oldTail = this->tail;
190
191 if (CompareAndSwap(&this->tail, oldTail, newNode))
192 break;
193 }
194
195 if (oldTail != nullptr)
196 oldTail->next = newNode;
197
198 // if head is null, set head to new node
199 CompareAndSwap(&this->head, nullptr, newNode);
200 CompareAndSwap(&this->tail, nullptr, newNode);
201}

206template<class TYPE>
207inline bool
208LockFreeQueue<TYPE>::Dequeue(TYPE& item)
209{
210 Node* oldHead;
211 Node* newHead;
212 while (true)
213 {
214 oldHead = this->head;
215 if (oldHead == nullptr)
216 return false;
217
218 newHead = oldHead->next;
219 if (CompareAndSwap(&this->head, oldHead, newHead))
220 break;
221 }
222
223 // set tail to nullptr if head became tail
224 oldHead->next = nullptr;
225
226 // copy return value here, if we dealloc and someone else allocs, they might have time to change the value
227 item = oldHead->data;
228
229 // make sure to return this node to the free list
230 this->Dealloc(oldHead);
231
232 int size = Interlocked::Decrement(&this->size);
233 n_assert(size >= 0);
234
235 return true;
236}
