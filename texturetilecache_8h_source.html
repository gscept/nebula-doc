129inline void
130TextureTileCache::Setup(uint tileSize, uint textureSize)
131{
132 this->head = this->tail = nullptr;
133 this->tiles = textureSize / tileSize;
134 this->tileSize = tileSize;
135
136 // keep all nodes linearly in memory!
137 this->nodes.Resize(this->tiles * this->tiles);
138
139 // setup storage
140 for (uint x = 0; x < this->tiles; x++)
141 {
142 for (uint y = 0; y < this->tiles; y++)
143 {
144 // calculate node index 2D->1D
145 uint index = x + y * this->tiles;
146
147 // set the data, this will be static
148 Node* node = &this->nodes[index];
149 node->offset = Math::uint2{ x * tileSize, y * tileSize };
150 node->entry = InvalidTileCacheEntry;
151 node->prev = nullptr;
152 node->next = nullptr;
153
154 // Add to linked list
155 this->InsertBeginning(node);
156 }
157 }

163inline TextureTileCache::CacheResult
164TextureTileCache::Cache(TileCacheEntry entry)
165{
166 n_assert(entry != InvalidTileCacheEntry);
167
168 CacheResult result;
169 result.didCache = false;
170 result.evicted = InvalidTileCacheEntry;
171
172 IndexT index = this->lookup.FindIndex(entry);
173 if (index == InvalidIndex)
174 {
175 // Grab the tail
176 Node* node = this->tail;
177 this->Remove(node);
178 if (node->entry != InvalidTileCacheEntry)
179 {
180 this->lookup.Erase(node->entry);
181 result.evicted = node->entry;
182 }
183
184 // make most recent
185 this->InsertBeginning(node);
186 node->entry = entry;
187 this->lookup.Add(entry, node);
188
189 // update entry
190 result.cached = node->offset;
191 result.didCache = true;
192 }
193 else
194 {
195 Node* node = this->lookup.ValueAtIndex(index);
196 result.cached = node->offset;
197 this->Remove(node);
198 this->InsertBeginning(node);
199 }
200 return result;
201}
