131inline void
132TextureTileCache::Setup(uint tileSize, uint textureSize, uint invalidValue)
133{
134 this->tiles = textureSize / tileSize;
135 this->tileSize = tileSize;
136 this->invalidValue = invalidValue;
137
138 // keep all nodes linearly in memory!
139 this->nodes.Resize(this->tiles * this->tiles);
140 this->nodePool.Reserve(this->tiles * this->tiles);
141
142 // setup storage
143 for (uint x = 0; x < this->tiles; x++)
144 {
145 for (uint y = 0; y < this->tiles; y++)
146 {
147 // calculate node index 2D->1D
148 uint index = x + y * this->tiles;
149
150 // set the data, this will be static
151 Node* node = &this->nodes[index];
152 node->offset = Math::uint2{ x * tileSize, y * tileSize };
153 node->entry = InvalidEntry;
154 node->prev = nullptr;
155 node->next = nullptr;
156 node->id = -1;
157 }
158 }
159
160 this->head = nullptr;
161 this->tail = nullptr;
162}

167inline bool
168TextureTileCache::Cache(TileCacheEntry entry, Math::uint2& coords)
169{
170 n_assert(entry != InvalidEntry);
171 IndexT index = this->lookup.FindIndex(entry);
172 if (index == InvalidIndex)
173 {
174 // if we have no free ids, free up tail
175 if (this->nodePool.GetNumFree() == 0)
176 {
177 this->nodePool.Dealloc(this->tail->id);
178 n_assert(this->tail->entry != InvalidEntry);
179
180 // also erase from lookup
181 this->lookup.Erase(this->tail->entry);
182 this->tail->id = -1;
183 this->tail->entry = InvalidEntry;
184 this->Remove(this->tail);
185 }
186
187 // allocate new node
188 uint nodeId = this->nodePool.Alloc();
189 Node* node = &this->nodes[nodeId];
190
191 // make most recent
192 this->InsertBeginning(node);
193 node->entry = entry;
194 node->id = nodeId;
195 this->lookup.Add(entry, node);
196
197 // update entry
198 coords = node->offset;
199 return true;
200 }
201 else
202 {
203 Node* node = this->lookup.ValueAtIndex(index);
204 this->Remove(node);
205 this->InsertBeginning(node);
206 return false;
207 }
208}

270TextureTileCache::Remove(Node* node)
271{
272 // reassign neighbors
273 if (node->prev == nullptr)
274 this->head = node->next;
275 else
276 node->prev->next = node->next;
277
278 if (node->next == nullptr)
279 this->tail = node->prev;
280 else
281 node->next->prev = node->prev;
282
283 node->next = nullptr;
284 node->prev = nullptr;
285}
