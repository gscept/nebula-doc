454inline void
455String::Alloc(SizeT newSize)
456{
457 n_assert(newSize > (this->strLen + 1));
458 n_assert(newSize > this->heapBufferSize);
459
460 // free old buffer
461 if (this->heapBuffer)
462 {
463 Memory::Free(Memory::StringDataHeap, (void*) this->heapBuffer);
464 this->heapBuffer = 0;
465 }
466
467 // allocate new buffer
468 this->heapBuffer = (char*) Memory::Alloc(Memory::StringDataHeap, newSize);
469 this->heapBufferSize = newSize;
470 this->localBuffer[0] = 0;

473//------------------------------------------------------------------------------
477inline void
478String::Realloc(SizeT newSize)
479{
480 n_assert(newSize > (this->strLen + 1));
481 n_assert(newSize > this->heapBufferSize);
482
483 // allocate a new buffer
484 char* newBuffer = (char*) Memory::Alloc(Memory::StringDataHeap, newSize);
485
486 // copy existing contents there...
487 if (this->strLen > 0)
488 {
489 const char* src = this->AsCharPtr();
490 Memory::Copy(src, newBuffer, this->strLen);
491 }
492 newBuffer[this->strLen] = 0;
493
494 // assign new buffer
495 if (this->heapBuffer)
496 {
497 Memory::Free(Memory::StringDataHeap, (void*) this->heapBuffer);
498 this->heapBuffer = 0;
499 }
500 this->localBuffer[0] = 0;
501 this->heapBuffer = newBuffer;
502 this->heapBufferSize = newSize;
503}
