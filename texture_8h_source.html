260inline TextureType
261TextureTypeFromString(const Util::String& string)
262{
263 if (string == "Texture1D") return Texture1D;
264 else if (string == "Texture2D") return Texture2D;
265 else if (string == "Texture3D") return Texture3D;
266 else if (string == "TextureCube") return TextureCube;
267 else if (string == "Texture1DArray") return Texture1DArray;
268 else if (string == "Texture2DArray") return Texture2DArray;
269 else if (string == "TextureCubeArray") return TextureCubeArray;
270 else
271 {
272 n_error("Unknown texture type '%s'", string.AsCharPtr());
273 return Texture1D;
274 }
275}

280inline TextureUsage
281TextureUsageFromString(const Util::String& string)
282{
283 Util::Array<Util::String> comps = string.Tokenize("|");
284 TextureUsage usage = TextureUsage(0x0);
285
286 for (IndexT i = 0; i < comps.Size(); i++)
287 {
288 if (comps[i] == "Sample") usage |= SampleTexture;
289 else if (comps[i] == "Render") usage |= RenderTexture;
290 else if (comps[i] == "ReadWrite") usage |= ReadWriteTexture;
291 else if (comps[i] == "TransferSource") usage |= TransferTextureSource;
292 else if (comps[i] == "TransferDestination") usage |= TransferTextureDestination;
293 }
294
295 return usage;
296}
