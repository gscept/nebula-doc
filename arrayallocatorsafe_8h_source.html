392ArrayAllocatorSafe<TYPES...>::Lock(const ArrayAllocatorAccess access)
393{
394 if (access == ArrayAllocatorAccess::Read)
395 {
396 this->lock.LockRead();
397 Threading::Interlocked::Increment(&this->numReaders);
398 }
399 else
400 {
401 this->lock.LockWrite();
402 Threading::Interlocked::Exchange(&this->writer, 1);
403 }
