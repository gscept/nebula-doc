42template<class T>
43struct AllocatorLock
44{
45 AllocatorLock(T* allocator, uint32_t element)
46 : allocator(allocator)
47 , element(element)
48 {
49 this->didAcquire = this->allocator->Acquire(this->element);
50 };
51
52 ~AllocatorLock()
53 {
54 if (this->didAcquire)
55 this->allocator->Release(this->element);
56 }
57
58 bool didAcquire;
59 uint32_t element;
60 T* allocator;
61};

409ArrayAllocatorSafe<MAX_ALLOCS, TYPES...>::TryAcquire(const uint32_t index)
410{
411 Threading::ThreadId myThread = Threading::Thread::GetMyThreadId();
412 Threading::ThreadId currentThread = Threading::Interlocked::CompareExchange((volatile int*)&this->owners[index], myThread, Threading::InvalidThreadId);
413 n_assert(currentThread == Threading::InvalidThreadId);

416//------------------------------------------------------------------------------
419template<uint MAX_ALLOCS, class ...TYPES>
420bool
421ArrayAllocatorSafe<MAX_ALLOCS, TYPES...>::Acquire(const uint32_t index)
422{
423 Threading::ThreadId myThread = Threading::Thread::GetMyThreadId();
424 if (this->owners[index] == myThread)
425 return false;
426
427 // Spinlock
428 while (Threading::Interlocked::CompareExchange((volatile int*)&this->owners[index], myThread, Threading::InvalidThreadId) != Threading::InvalidThreadId)
429 {
430 Threading::Thread::YieldThread();
431 };
432 return true;
433}
