73 static uint32_t TypeToSize(const Type type)
74 {
75 switch (type)
76 {
77 case Type::TextureHandle:
78 return sizeof(TextureHandleTuple);
79 break;
80 case Type::Float:

83 case Type::Vec2:

86 case Type::Vec4:

88 break;
89 case Type::Int:
90 return sizeof(int32);
91 break;
92 case Type::UInt:
93 return sizeof(uint32);
94 break;
95 case Type::Bool:
